import React from 'react'
import PropTypes from 'prop-types'
import Tooltip from '../atoms/Tooltip'
import { IconError } from '../atoms/Icons'
 
class Input extends React.Component {
  constructor(props) {
    super(props)
    this.state = { tooltipVisible: this.props.tooltipVisible }
    this.tooltipOn = this.tooltipOn.bind(this)
    this.tooltipOff = this.tooltipOff.bind(this)
  }
  tooltipOn() {
    this.setState({ tooltipVisible: true })
  }
  tooltipOff() {
    this.setState({ tooltipVisible: false })
  }
 
  render() {
    let classNames = 'form-control'
    let wrapperClasses = 'input-wrapper'
    if (this.props.done) classNames += ' done'
    if (this.props.alert) {
      classNames += ' alert'
      wrapperClasses += ' has-feedback'
    }
 
    return (
      <div className="form-group">
        {this.props.label && <label htmlFor={this.props.label}>{this.props.label}</label>}
        <span className={wrapperClasses}>
          <input
            type={this.props.type}
            name={this.props.name}
            className={classNames}
            id={this.props.label}
            data-test={this.props.label}
            placeholder={this.props.placeholder}
            onChange={this.props.onChange}
            onKeyUp={this.props.onKeyUp}
            onFocus={this.tooltipOn}
            onBlur={this.tooltipOff}
            value={this.props.value}
            required={this.props.required}
            autoFocus={this.props.autoFocus}
          />
          {this.props.alert && (
            <div>
              <Tooltip
                text={this.props.tooltipText}
                visible={this.state.tooltipVisible}
              />
              <IconError className="error form-control-feedback" />
              <span className="tooltip-text form-control-feedback">View help.</span>
            </div>
          )}
        </span>
      </div>
    )
  }
}
 
Input.defaultProps = {
  type: 'text',
  required: false,
}
 
Input.propTypes = {
  name: PropTypes.string,
  label: PropTypes.string,
  placeholder: PropTypes.string,
  value: PropTypes.string,
  required: PropTypes.bool,
  autoFocus: PropTypes.bool,
  onChange: PropTypes.func,
  onKeyUp: PropTypes.func,
  type: PropTypes.string,
  done: PropTypes.bool,
  alert: PropTypes.bool,
  tooltipText: PropTypes.string,
  tooltipVisible: PropTypes.bool,
}
 
export default Input
