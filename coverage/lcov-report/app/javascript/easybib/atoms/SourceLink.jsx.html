import React from 'react'
import PropTypes from 'prop-types'
 
function SourceLink({ source }) {
  const getLink = () => {
    switch (source.data.source) {
      case 'website':
      case 'webpage':
        if (source.data.autocite && source.data.autocite.url) {
          return <a href={source.data.autocite.url}>{source.data.autocite.url}</a>
        }
        return <a href={source.display.displayurl}>{source.display.displayurl}</a>
      case 'newspaper':
        return <a href={source.display.url}>{source.display.url}</a>
      default:
        if (source.link)
          return (
            <a target="_blank" rel="noopener noreferrer" href={source.link}>
              Find at my library
            </a>
          )
 
        return ''
    }
  }
 
  return <span>{getLink()}</span>
}
 
SourceLink.propTypes = {
  source: PropTypes.shape({
    link: PropTypes.string,
    display: PropTypes.shape({
      displayurl: PropTypes.string,
    }),
    data: PropTypes.shape({
      autocite: PropTypes.shape({
        url: PropTypes.string,
      }),
    }),
  }),
}
 
export default SourceLink
