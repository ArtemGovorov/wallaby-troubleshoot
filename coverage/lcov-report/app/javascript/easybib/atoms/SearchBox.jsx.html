import React, { Component } from 'react'
import PropTypes from 'prop-types'
import { IconSearch } from '../atoms/Icons'
 
class SearchBox extends Component {
  constructor(props) {
    super(props)
    this.state = { inputText: '' }
  }
  componentWillUpdate(nextProps) {
    if (this.props.reset === false && nextProps.reset === true)
      // eslint-disable-next-line react/no-will-update-set-state
      this.setState({ inputText: '' })
  }
  componentDidUpdate() {
    // focus the search input box when we open it
    this.searchInput.focus()
  }
 
  handleChange(e) {
    const term = e.target.value
    this.setState({ inputText: term })
    if (this.props.manualSearch) {
      return
    }
    if (term === '') {
      if (this.timeout) clearTimeout(this.timeout)
      if (this.props.cancelSearch) this.props.cancelSearch()
      return
    }
    if (this.timeout) clearTimeout(this.timeout)
    this.timeout = setTimeout(() => {
      this.props.handleSearch(term)
    }, 800)
  }
 
  handleKeyUp(e) {
    if (e.which === 13 && this.state.inputText !== '')
      this.props.handleSearch(this.state.inputText)
  }
 
  render() {
    const inputType = this.props.manualSearch ? 'search' : 'text'
    return (
      <span className={this.props.mainClass}>
        <input
          className={this.props.inputClass}
          type={inputType}
          placeholder={this.props.placeholder}
          onChange={e => this.handleChange(e)}
          onKeyUp={e => this.handleKeyUp(e)}
          value={this.state.inputText}
          ref={input => {
            this.searchInput = input
          }}
        />
        <IconSearch className="search-icon pull-right" />
        {this.props.searching && <div className={this.props.searchingClass} />}
      </span>
    )
  }
}
 
SearchBox.propTypes = {
  handleSearch: PropTypes.func.isRequired,
  cancelSearch: PropTypes.func, // eslint-disable-line react/no-unused-prop-types
  searching: PropTypes.bool,
  manualSearch: PropTypes.bool,
  reset: PropTypes.bool,
  mainClass: PropTypes.string,
  inputClass: PropTypes.string,
  searchingClass: PropTypes.string,
  placeholder: PropTypes.string,
}
 
export default SearchBox
