import React, { Component } from 'react'
import PropTypes from 'prop-types'
 
function ButtonBase({
  theme,
  size = 'md',
  type,
  extraCss,
  right,
  dataId,
  gtmAction,
  children,
  ...props
}) {
  let buttonClasses = 'btn'
  buttonClasses += theme ? ` ${theme}` : ''
  buttonClasses += size ? ` ${size}` : ''
  buttonClasses += extraCss ? ` ${extraCss}` : ''
  buttonClasses += right ? ' pull-right' : ''
 
  return (
    <button {...props} type={type} data-id={dataId} className={buttonClasses}>
      {children}
    </button>
  )
}
 
ButtonBase.propTypes = {
  theme: PropTypes.oneOf(['primary', 'secondary', 'tertiary', 'one', 'two']),
  size: PropTypes.oneOf(['sm', 'md', 'lg']),
  right: PropTypes.bool,
  type: PropTypes.oneOf(['button', 'submit']),
  extraCss: PropTypes.string,
  onClick: PropTypes.func,
  disabled: PropTypes.bool,
  dataId: PropTypes.string,
  gtmAction: PropTypes.string,
  children: PropTypes.node,
}
 
class Button extends Component {
  constructor(props) {
    super(props)
    this.state = { disabled: this.props.disabled }
  }
 
  componentWillUpdate(nextProps) {
    if (nextProps.disabled !== this.state.disabled)
      // eslint-disable-next-line react/no-will-update-set-state
      this.setState({ disabled: nextProps.disabled })
  }
 
  handleClick = e => {
    if (this.props.onePress) {
      if (!this.state.disabled) {
        if (this.props.onClick) this.props.onClick(e)
        this.setState({ disabled: true })
      }
    } else if (this.props.onClick) {
      this.props.onClick(e)
    }
  }
 
  render() {
    const { disabled, children, onClick, onePress, ...props } = this.props
 
    return (
      <ButtonBase disabled={this.state.disabled} onClick={this.handleClick} {...props}>
        {children}
      </ButtonBase>
    )
  }
}
 
Button.propTypes = ButtonBase.propTypes
Button.onePress = PropTypes.bool
 
export default Button
