import React, { Component } from 'react'
import PropTypes from 'prop-types'
import onClickOutside from 'react-onclickoutside'
import uuidV1 from 'uuid'
import { SelectedButton } from 'easybib/pages/Bibliography/styles'
import Icon from 'easybib/components/Icon'
import styled from 'styled-components'
import { IconExpandMore } from '../atoms/Icons'
 
const IconContainer = styled.div`
  width: 18px;
  top: 4px;
  position: absolute;
  display: inline;
`
 
const DropdownButton = styled(SelectedButton)`
  padding-right: 6px;
  padding-left: 0;
`
 
class Dropdown extends Component {
  constructor(props) {
    super(props)
    this.handleClickOutside = this.handleClickOutside.bind(this)
    this.handleSelect = this.handleSelect.bind(this)
    this.handleToggle = this.handleToggle.bind(this)
    this.state = { open: false }
  }
 
  handleClickOutside() {
    this.setState({ open: false })
  }
 
  handleSelect(args) {
    this.setState({ open: false })
    this.props.onSelect(args)
  }
 
  handleToggle(_, handleClick) {
    this.setState({ open: !this.state.open })
    if (handleClick) handleClick()
  }
 
  render() {
    let buttonClassName = 'btn btn-default dropdown-toggle'
    let aClassName = 'btn btn-default dropdown-toggle'
    let wrapperClassName = 'dropdown links'
    const { params } = this.props
 
    if (this.props.buttonCss) {
      buttonClassName = `${buttonClassName} ${this.props.buttonCss}`
      aClassName = `${aClassName} ${this.props.buttonCss}`
    }
    if (this.props.wrapperCss) {
      wrapperClassName = `${wrapperClassName} ${this.props.wrapperCss}`
    }
    if (this.state.open) {
      wrapperClassName += ' open'
    }
    const target = this.props.newWindow ? '_blank' : ''
 
    const getLink = url => (params ? `${url}${params}` : url)
 
    return (
      <nav className={wrapperClassName}>
        {this.props.link && (
          <a
            className={aClassName}
            target={target}
            href={this.props.link}
            onClick={this.props.onClick}
          >
            {this.props.name}
          </a>
        )}
        {!this.props.link &&
          !this.props.newButton && (
            <button
              aria-haspopup
              className={buttonClassName}
              id={this.props.name}
              type="button"
              onClick={this.handleToggle}
            >
              {this.props.name} <IconExpandMore size={15} color="#333" />
            </button>
          )}
        {!this.props.link &&
          this.props.newButton && (
            <DropdownButton
              primary
              aria-haspopup
              id={this.props.name}
              type="button"
              onClick={this.handleToggle}
            >
              {this.props.name}{' '}
              <IconContainer>
                <Icon
                  name="downArrow"
                  className="pull-right down-arrow-icon material-icons"
                  size="15px"
                />
              </IconContainer>
            </DropdownButton>
          )}
        {this.props.items &&
          this.props.items.length && (
            <ul className="items" aria-labelledby={this.props.name}>
              {this.props.items.map(item => (
                <li key={uuidV1()}>
                  {!this.props.onSelect && (
                    <a
                      href={item.link && getLink(item.link)}
                      target={target}
                      title={item.title}
                      onClick={e => this.handleToggle(e, item.onClick)}
                    >
                      {item.name}
                    </a>
                  )}
                  {this.props.onSelect && (
                    <button onClick={() => this.handleSelect(item.value)} type="button">
                      {item.name}
                    </button>
                  )}
                </li>
              ))}
            </ul>
          )}
      </nav>
    )
  }
}
 
Dropdown.propTypes = {
  name: PropTypes.string.isRequired,
  link: PropTypes.string,
  buttonCss: PropTypes.string,
  wrapperCss: PropTypes.string,
  items: PropTypes.arrayOf(
    PropTypes.shape({
      name: PropTypes.string.isRequired,
      link: PropTypes.string,
      title: PropTypes.string,
      onClick: PropTypes.func,
    })
  ),
  params: PropTypes.string,
  onSelect: PropTypes.func,
  newWindow: PropTypes.bool,
  onClick: PropTypes.func,
  newButton: PropTypes.bool,
}
 
export default onClickOutside(Dropdown)
