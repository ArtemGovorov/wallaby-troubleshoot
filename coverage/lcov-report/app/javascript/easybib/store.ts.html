import { createStore, applyMiddleware, compose, StoreCreator, Store } from 'redux'
import { createLogger } from 'redux-logger'

import thunk from 'redux-thunk'
import reducers from 'easybib/reducers'
import isWindowDefined from 'easybib/utils/isWindowDefined'
 
import { EasybibState } from 'types/easybibRedux'
 
import * as actionCreators from './actions/actions.types'
import { dataApi, formatApi, sourceApi, tokenStorage } from './configureApi'

const extras = { dataApi, formatApi, sourceApi, tokenStorage }

const enhancers = [applyMiddleware(thunk.withExtraArgument(extras))]

// We do this because we want the ability to import store multiple times,
// but only actually be configured ONCE
 
let store: Store<EasybibState> // eslint-disable-line import/no-mutable-exports
 
if (!store) {
  if (process.env.NODE_ENV !== 'production') {
    const logger = createLogger({ collapsed: true })
    enhancers.push(applyMiddleware(logger))
    if (isWindowDefined) {
      if (window.devToolsExtension) {
        enhancers.push(window.devToolsExtension({ actionCreators }))
      }
    }
  }

  const parsedStore =
    isWindowDefined && window.gon ? JSON.parse(window.gon.state) : undefined
 
  const configureStore = compose(...enhancers)(createStore) as StoreCreator
  store = configureStore(reducers, parsedStore)
}

export default store
