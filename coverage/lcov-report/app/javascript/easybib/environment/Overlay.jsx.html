import React, { Component } from 'react'
import PropTypes from 'prop-types'
import ClickOutside from '../molecules/ClickOutside'
import { getLocalBibInfo, setLocal } from '../utils/localStorage'
 
/**
 * Takes a selector as a string, grabs the element, and adds a class
 * @param {string} selector
 * @return {Element}
 */
const highlight = selector => {
  const element = document.querySelector(selector)
  if (element) element.classList.add('focus')
  return element
}
 
/**
 * Takes a selector as a string, grabs the element, and removes a class
 * @param {string} selector
 */
const unhighlight = selector => {
  const element = document.querySelector(selector)
  if (element) element.classList.remove('focus')
}
 
/**
 * Conditionally increments the overlay page index if a page
 * has not been visited yet. Leverages local storage to maintain
 * record of whether page has been accessed or not already.
 * If page isn't found, create new array based on previous array
 * in local storage, with new page added
 * @param {string} page - canonical name of the page function is used on
 */
const overlayFirstVisited = pageName =>
  new Promise(resolve => {
    const visited = getLocalBibInfo().pagesVisited || []
    const pageVisited = visited.includes(pageName)
    if (!pageVisited) {
      visited.push(pageName)
      setLocal('pagesVisited', visited)
      resolve()
    }
  })
 
const propTypes = {
  page: PropTypes.string,
  steps: PropTypes.arrayOf(PropTypes.object),
}
 
class OverlayComponent extends Component {
  constructor() {
    super()
    this.handleClick = this.handleClick.bind(this)
    this.focusWrapper = this.focusWrapper.bind(this)
    this.currentStep = this.currentStep.bind(this)
    this.state = {
      visible: false,
      currentIndex: 0,
    }
  }
 
  componentDidMount() {
    overlayFirstVisited(this.props.page).then(() => {
      this.showOverlay()
      this.focusCurrentIndex()
    })
  }
 
  currentStep() {
    return this.props.steps[this.state.currentIndex]
  }
 
  /**
   * Attempts to highlight currentIndex.
   */
  focusCurrentIndex() {
    const selectorFound = highlight(this.currentStep().selector)
 
    if (!selectorFound) this.focusWrapper()
  }
 
  /**
   * Handles unhighlighting, incrementing, and hiding of overlay
   */
  focusWrapper() {
    unhighlight(this.currentStep().selector)
    if (this.state.currentIndex < this.props.steps.length - 1) {
      this.incrementIndex()
      this.focusCurrentIndex()
    } else {
      // If there aren't any more items in the list, hide the overlay
      this.hideOverlay()
    }
  }
 
  showOverlay() {
    this.setState({ visible: true })
  }
 
  hideOverlay() {
    this.setState({ visible: false })
  }
 
  incrementIndex() {
    this.setState({
      currentIndex: this.state.currentIndex + 1,
    })
  }
 
  handleClick() {
    this.focusWrapper()
  }
 
  render() {
    const className = this.currentStep().class || null
    return (
      this.state.visible && (
        <span>
          <ClickOutside onClickOutside={this.handleClick}>
            <span />
          </ClickOutside>
          <div key="A" className={`overlay ${className}`} />
          <div key="B" className="overlay cta-overlay" />
        </span>
      )
    )
  }
}
 
OverlayComponent.propTypes = propTypes
 
export default OverlayComponent
