import { combineReducers } from 'redux'

import { loadingReducer } from '@chegg/redux-loading-queue'
import citationsReducer from 'easybib/reducers/citations'
import citationReducer from 'easybib/reducers/citation'
import formReducer from 'easybib/reducers/form'
import userReducer from 'easybib/reducers/user'
import stylesReducer from 'easybib/reducers/styles'
import stylesSearchReducer from 'easybib/reducers/stylesSearch'
import typesReducer from 'easybib/reducers/types'
import sourcesReducer from 'easybib/reducers/sources'
import searchReducer from 'easybib/reducers/search'
import modalReducer from 'easybib/reducers/modal'
import errorReducer from 'easybib/reducers/error'
import pageNotFoundReducer from 'easybib/reducers/pageNotFound'
import showProPaywallReducer from 'easybib/reducers/showProPaywall'
import appReducer from 'easybib/reducers/app'
import searchDataReducer from 'easybib/reducers/searchData'
import projectIDReducer from 'easybib/reducers/projectID'
import currentSourceTypeReducer from 'easybib/reducers/currentSourceType'
import cellReducer from 'easybib/reducers/cell'

import currentProjectReducer from 'easybib/reducers/currentProject'
import adLoaderReducer from 'easybib/reducers/adLoader'
 
// Selectors TODO: move these into a separate file?
export const getCitationStyle = state => state.currentCitationStyle
export const getSourceType = state => state.currentSourceType
export const getCurrentProjectID = state => state.currentProject
export const getProjects = state => state.projects
export const getCitations = state => state.citations
export const getSelectedCitationID = state => state.citation
 
export const getSourceTypeTitle = state => {
  const typeMatch = state.types.find(
    stateType => stateType.value === getSourceType(state)
  )
  return typeMatch ? typeMatch.title : null
}
 
export const getProject = (state, projectID) => {
  const projects = getProjects(state)
  return projects.find(project => project.id === projectID)
}
 
export const getCurrentProject = state => {
  const currentProjectID = getCurrentProjectID(state)
  return getProject(state, currentProjectID)
}
 
export const getSelectedCitation = state => {
  const selectedCitationID = getSelectedCitationID(state)
  const citations = getCitations(state)
  return citations.find(citation => citation.id === selectedCitationID)
}
 
export const getSelectedCitations = state => {
  const citations = getCitations(state)
  return citations.filter(citation => citation.selected)
}
 
const reducers = combineReducers({
  app: appReducer, // All pages
  projectID: projectIDReducer,
  currentCitationStyle: currentCitationStyleReducer,
  currentSourceType: currentSourceTypeReducer,
  currentProject: currentProjectReducer,
  citation: citationReducer, // FormPage, EvalPage, BibliographyPage
  citations: citationsReducer, // BibliographyPage
  projects: projectsReducer,
  error: errorReducer,
  pageNotFound: pageNotFoundReducer,
  form: formReducer,
  loading: loadingReducer,
  modal: modalReducer,
  showProPaywall: showProPaywallReducer,
  search: searchReducer,
  searchData: searchDataReducer,
  adLoader: adLoaderReducer,
  sources: sourcesReducer,
  styles: stylesReducer,
  stylesSearch: stylesSearchReducer,
  types: typesReducer,
  user: userReducer,
  cell: cellReducer,
})
 
export default reducers
