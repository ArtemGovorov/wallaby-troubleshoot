import ActionTypes from 'easybib/actions/actions.types'
 
export const citationsReducer = (state = [], action) => {
  switch (action.type) {
    case ActionTypes.CITATIONS_LOAD:
      return [...action.citations]
    case ActionTypes.CITATION_ADD:
      return [...state, action.citation]
    case ActionTypes.CITATION_REMOVE:
      return state.filter(citation => citation.id !== action.id)
    case ActionTypes.CITATION_UPDATE:
      return state.map(citation => {
        if (citation.id === action.id) {
          return { ...citation, ...action.data }
        }
        return citation
      })
    default:
      return state
  }
}
 
export default citationsReducer
