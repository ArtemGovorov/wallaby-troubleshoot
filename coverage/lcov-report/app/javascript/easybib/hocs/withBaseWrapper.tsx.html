import React from 'react'
import Advert from 'easybib/atoms/Advert'
import AppHeader from 'easybib/components/AppHeader'
import Footer from 'easybib/molecules/Footer'
import styled from 'styled-components'

import initGTM from 'easybib/utils/initGTM'
import initHotjar from 'easybib/utils/initHotjar'
 
import withThemeProvider from './withThemeProvider'
import withReduxProvider from './withReduxProvider'

const Main = styled.main`
  border-top: 1px solid #ededed;
`

interface CenterNodeProps {
  children: JSX.Element
}

const CenterNode: React.SFC<CenterNodeProps> = props => (
  <main id="easybib-content">
    <div className="eb-site-container">
      <Advert placement="T1" className="adds top" />
      <div className="eb-container">
        <div className="adds left">
          <Advert placement="L1" />
          <Advert placement="L2" />
        </div>
        <div className="center">
          <Main id="main-content">{props.children}</Main>
        </div>
        <div className="adds right">
          <Advert placement="R1" />
          <Advert placement="R2" />
          <Advert placement="R3" />
        </div>
      </div>
    </div>
  </main>
)

interface WrappedProps {
  url: string
}

function withBaseWrapper(Component, initAction?) {
  class Wrapped extends React.Component<WrappedProps> {
    componentDidMount() {
      initGTM()
      initHotjar()
    }
 
    render() {
      return (
        <React.Fragment>
          <AppHeader withOutRouter />
          <CenterNode>
            <Component {...this.props} />
          </CenterNode>
          <div id="easybib-bottom-add" />
          <footer id="easybib_footer">
            <Footer />
            <Advert placement="D_SKIN" />
            <Advert placement="D_SB" />
            <Advert placement="D_MISC" />
          </footer>
        </React.Fragment>
      )
    }
  }

  return withThemeProvider(withReduxProvider(Wrapped, initAction))
}
 
export default withBaseWrapper
