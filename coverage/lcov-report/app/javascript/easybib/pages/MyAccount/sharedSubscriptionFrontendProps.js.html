import isWindowDefined from 'easybib/utils/isWindowDefined'
import environment from 'easybib/utils/environment'

function getCookie(cookies, key) {
  const match = cookies.match(new RegExp(`(^| )${key}=([^;]+)`)) || []
  const [, , val] = match
  return val
}
 
let cheggAuthToken = ''
let apiUrl = ''
if (isWindowDefined && process.env.NODE_ENV !== 'test') {
  const config = window.gon.subscription_frontend_config

  if (!config) {
    throw new Error(
      'subscription-frontend cannot run because the `subscription_frontend_config` property is missing from `window.gon`.'
    )
  }
  cheggAuthToken = getCookie(document.cookie, config.cheggAuthIdTokenCookie)
  apiUrl = config.subApiUrl
}

const variant = 'ecom' // This will not vary in CFE -- always ecom variant.

export default {
  apiUrl,
  cheggAuthToken,
  variant,
  deployEnvironment: environment,
}
