import React from 'react'
import PropTypes from 'prop-types'
import uuidV1 from 'uuid'
 
import Table from 'easybib/components/Table/Base'

import Row from './Row'

const uuid = index => {
  if (typeof jest === 'undefined') {
    return uuidV1()
  }
  return `abcdefg${index}`
}
 
const PlanTable = ({ rowsProps }) => {
  const rows = rowsProps.map((props, index) => <Row {...props} key={uuid(index)} />)

  return (
    <Table>
      <tbody>
        <tr>
          <th>Plan name</th>
          <th>Plan description</th>
          <th>Billing Details</th>
          <th>Status</th>
        </tr>
        {rows}
      </tbody>
    </Table>
  )
}

PlanTable.propTypes = {
  rowsProps: PropTypes.arrayOf(PropTypes.shape(Row.propTypes)).isRequired,
}

export default PlanTable
