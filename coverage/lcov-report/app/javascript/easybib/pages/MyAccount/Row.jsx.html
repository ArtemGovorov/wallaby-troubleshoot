import React from 'react'
import PropTypes from 'prop-types'
import style from 'styled-components'
import Button from 'easybib/components/Button'
import Link from 'easybib/components/Link'
import Cell from 'easybib/components/Table/Cell'
 
const MaxWidth = style.div`
  max-width: 300px;
`
 
const Row = ({ name, desc, priceFrequency, status, initCancel, expirationDate }) => {
  const dateOrDefault = expirationDate || 'at the end of this billing period'
  let statusCell
  switch (status) {
    case 'active':
      statusCell = (
        <MaxWidth>
          Currently active.{' '}
          {initCancel && (
            <Link gray pointer onClick={initCancel}>
              Cancel?
            </Link>
          )}
        </MaxWidth>
      )
      break
    case 'pendingCancel':
      statusCell = (
        <MaxWidth>
          <p>We are sorry to see you go!</p>
          <p>
            {' '}
            Your subscription is cancelled effective {dateOrDefault}. You'll have access
            to all Plus features until then.
          </p>
          <p>Please, let us know how we can improve.</p>
          <Button
            newPrimary
            size="lg"
            href="https://in.hotjar.com/s?siteId=7064&surveyId=110173"
            target="_blank"
          >
            Submit Feedback
          </Button>
        </MaxWidth>
      )
      break
    case 'cancelled':
      statusCell = 'Cancelled.'
      break
    default:
      statusCell = ''
  }
 
  return (
    <tr key={`${name}-${status}`}>
      <Cell highlight pad>
        {name}
      </Cell>
      <Cell highlight pad>
        {desc}
      </Cell>
      <Cell highlight pad>
        {priceFrequency}
      </Cell>
      <Cell highlight pad>
        {statusCell}
      </Cell>
    </tr>
  )
}
 
Row.propTypes = {
  name: PropTypes.string.isRequired,
  desc: PropTypes.string.isRequired,
  priceFrequency: PropTypes.string.isRequired,
  initCancel: PropTypes.func,
  status: PropTypes.string.isRequired,
  expirationDate: PropTypes.string.isRequired,
}
 
export default Row
