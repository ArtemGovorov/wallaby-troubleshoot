import React from 'react'
import { EasybibState } from 'types/easybibRedux'
import { connect } from 'react-redux'
import { Dispatch, AnyAction } from 'redux'
import { styleEvent } from 'easybib/utils/gtm'
import { setCitationStyle } from 'easybib/actions/actions.shared'
import { nonProRedirect } from 'easybib/actions/actions.redirects'
import StylesDropdown from 'easybib/molecules/StylesDropdown'
import { getCitationStyle } from 'easybib/reducers'
import { POPULAR_STYLES } from 'easybib/constants'
import Label from 'easybib/components/Form/Label'

interface PopularStylesSelectProps {
  selected: string
  onChange: (value: string) => void
}
 
export const PopularStylesSelect: React.SFC<PopularStylesSelectProps> = props => (
  <section className="citation-nav">
    <div className="switch-field">
      <div className="citations-radios visible-sm visible-md visible-lg">
        {POPULAR_STYLES.map(option => (
          <div key={option.value}>
            <input
              type="radio"
              id={`citation-style-radio-${option.value}`}
              value={option.value}
              onChange={e => props.onChange(e.target.value)}
              checked={option.value === props.selected}
              name="citation-style-radio"
              className="sr-only"
            />
            <label htmlFor={`citation-style-radio-${option.value}`}>
              <span className="radio-span">{option.name}</span>
            </label>
          </div>
        ))}
      </div>
      <div className="visible-xs">
        <Label>Style type</Label>
      </div>
      <StylesDropdown className="citation-style" />
    </div>
  </section>
)
 
const mapStateToProps = (state: EasybibState) => ({
  selected: getCitationStyle(state),
})
 
export const mapDispatchToProps = (dispatch: Dispatch<AnyAction>) => ({
  onChange: (style: string) => {
    if (style !== 'mla7' && style !== 'mla8') dispatch(nonProRedirect())
    dispatch(setCitationStyle(style))
    styleEvent(style)
  },
})
 
export default connect(
  mapStateToProps,
  mapDispatchToProps
)(PopularStylesSelect)
