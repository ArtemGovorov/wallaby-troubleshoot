import React from 'react'
import { connect } from 'react-redux'
import { EasybibState } from 'types/easybibRedux'
import Advert from 'easybib/atoms/Advert'
import { userIsPro } from 'easybib/actions/actions.redirects'
import SearchCitation from 'easybib/pages/Style/SearchCitation'
import UnlockProAdvert from 'easybib/pages/Style/UnlockProAdvert'

interface SearchWidgetProps {
  isPro?: boolean
}

export const SearchWidget: React.SFC<SearchWidgetProps> = ({ isPro }) => (
  <React.Fragment>
    <div className="home-page container-fluid">
      <div className="main-header">
        <h1>Start making citations the EasyBib way</h1>
        <span>Make citations in MLA, APA, Chicago, and 7000+ styles</span>
      </div>
      <div className="overlay" />
    </div>
    <SearchCitation />
    <Advert placement="C1" />
    {!isPro && <UnlockProAdvert />}
  </React.Fragment>
)
 
const mapStateToProps = (state: EasybibState) => ({
  isPro: userIsPro(state.user.role),
})
 
export default connect(mapStateToProps)(SearchWidget)
