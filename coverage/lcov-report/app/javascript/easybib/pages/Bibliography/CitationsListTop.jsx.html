import React, { Component } from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
import { deleteCitation } from 'redux-citation-api'
import { getCitationStyle, getSelectedCitations } from 'easybib/reducers'
import clipboard from 'easybib/utils/copypaste'
import { FlexRow } from 'easybib/components/Flex'
import StylesDropdown from 'easybib/molecules/StylesDropdown'
import Dropdown from 'easybib/atoms/Dropdown'
import RoundNumber from 'easybib/atoms/RoundNumber'
import Config from 'easybib/config'
import { sendSaveEvent, copyAllEvent } from 'easybib/utils/gtm'
import {
  gtmSortCitationsEvent,
  loadCitations,
  showCitationModals,
} from 'easybib/actions/actions.bibliography'
import CheckboxContainer from 'easybib/components/CheckboxContainer'
import { sortCitations } from 'easybib/utils/utils'
import {
  SelectedButtonContainer,
  ProjectName,
  SortContainer,
  StylesDropdownContainer,
  SelectedButton,
  SelectContainer,
  OrangeLabel,
  Label,
} from './styles'

const getLink = link => `${Config.easybibLegacyUrl}/${link}`

const saveMenu = {
  name: 'Save',
  link: getLink('register'),
}
 
const sortMenu = {
  name: 'Sort',
  menuItems: [
    {
      name: 'Alphabetical',
      link: '',
      value: 'alphabetical',
    },
    {
      name: 'Least recent',
      link: '',
      value: 'dateLeastRecent',
    },
    {
      name: 'Most recent',
      link: '',
      value: 'dateMostRecent',
    },
  ],
}

const shareMenu = {
  name: 'Share',
  items: [
    {
      name: 'Email Bibliography',
      link: getLink('export/docx/email'),
      title: 'Get a unique URL to share your bibliography',
    },
    {
      name: 'Share public link',
      link: getLink('public/share'),
    },
  ],
}
 
const citationsPlainText = items =>
  items
    .map(item => `${(item.formatted_data.output || '').replace(/(<([^>]+)>)/gi, '')}\n`)
    .join('')
 
const citationsHtml = items =>
  items
    .map(
      item =>
        `<p style="margin-left: 1cm; text-indent: -1cm;">${item.formatted_data.output ||
          ''}</p>`
    )
    .join('')
 
const propTypes = {
  user: PropTypes.shape({
    loggedIn: PropTypes.bool,
  }),
  selectedCitations: PropTypes.arrayOf(
    PropTypes.shape({
      text: PropTypes.string,
      type: PropTypes.string,
    })
  ),
  citations: PropTypes.arrayOf(
    PropTypes.shape({
      text: PropTypes.string,
      type: PropTypes.string,
    })
  ),
  projectId: PropTypes.string,
  showSave: PropTypes.bool,
  header: PropTypes.string.isRequired,
  handleSortChange: PropTypes.func,
  projectName: PropTypes.string,
  handleFetchProjects: PropTypes.func,
  updateCitations: PropTypes.func,
  handleDelete: PropTypes.func,
}
 
class CitationsListTop extends Component {
  constructor(props) {
    super(props)
    this.state = { checked: false }
    this.exportSelectedMenu = [
      {
        name: 'Copy to Clipboard',
        onClick: () => this.copyAll(true),
      },
      {
        name: 'Copy to Another Project',
        onClick: () => {
          this.props.handleFetchProjects()
          this.handleCheck({ deselect: true })
        },
      },
    ]
  }
 
  getParams = () => `?bibliography_id=${this.props.projectId}`

  exportMenu = () => {
    const menu = {
      name: 'Export List',
      items: [
        {
          name: 'Export to Word Doc',
          link: getLink('export/docx/index'),
          title: 'Your bibliography will be formatted for Word',
        },
        {
          name: 'Save to Google Docs',
          link: getLink('export/docx/gdocs'),
          title: 'Your bibliography will be saved in Google Docs',
        },
        {
          name: 'Save to Dropbox',
          link: getLink('export/docx/dropbox'),
          title: 'Your bibliography will be saved to Dropbox',
        },
        {
          name: 'Save to One Drive',
          link: getLink('export/docx/onedrive'),
          title: 'Your bibliography will be saved to OneDrive',
        },
        {
          name: 'Email Bibliography',
          link: getLink('export/docx/email'),
          title: 'Email your bibliography to yourself or others',
        },
        {
          name: 'Copy & Paste',
          link: getLink('export/docx/html'),
        },
      ],
    }
    if (this.props.user.loggedIn) {
      menu.items.push({
        name: 'Copy All to Another Project',
        onClick: this.props.handleFetchProjects,
      })
    }
    return menu
  }
 
  handleCheck = ({ deselect }) => {
    const updatedCitations = this.props.citations.map(citation => {
      const newCitation = { ...citation }
      newCitation.selected = deselect ? false : !this.state.checked
      return newCitation
    })
    this.props.updateCitations(updatedCitations)
    this.setState(state => ({
      checked: deselect ? false : !state.checked,
    }))
  }
 
  deleteSelected = () => {
    this.props.selectedCitations.forEach(citation => this.props.handleDelete(citation))
  }

  copyAll = onlySelected => {
    const citations = onlySelected ? this.props.selectedCitations : this.props.citations
    if (!onlySelected) copyAllEvent()
    clipboard.copy({
      'text/plain': citationsPlainText(citations),
      'text/html': citationsHtml(citations),
    })
    this.handleCheck({ deselect: true })
  }

  render() {
    return (
      <span>
        <div className="citations center-children">
          <div>
            <div className="citations-tools center-children">
              <div className="list-actions">
                <FlexRow>
                  {this.props.showSave && (
                    <Dropdown
                      name={saveMenu.name}
                      link={saveMenu.link}
                      items={saveMenu.items}
                      onClick={sendSaveEvent}
                    />
                  )}
                  {this.props.projectId && (
                    <React.Fragment>
                      <Dropdown
                        name={shareMenu.name}
                        link={shareMenu.link}
                        items={shareMenu.items}
                        params={this.getParams()}
                        newWindow
                      />
                      <Dropdown
                        name={this.exportMenu().name}
                        link={this.exportMenu().link}
                        items={this.exportMenu().items}
                        params={this.getParams()}
                        newWindow
                      />
                    </React.Fragment>
                  )}
                  {this.props.citations.length > 1 && (
                    <a className="btn btn-default dropdown-toggle" onClick={this.copyAll}>
                      Copy All to Clipboard
                    </a>
                  )}
                  <StylesDropdownContainer>
                    <StylesDropdown className="btn-group small-style" page="project" />
                  </StylesDropdownContainer>
                </FlexRow>
              </div>
              <FlexRow className="list-nav">
                <FlexRow>
                  <h1>
                    {this.props.header}{' '}
                    {this.props.citations.length > 0 && (
                      <RoundNumber number={this.props.citations.length} />
                    )}
                  </h1>
                  <ProjectName>{this.props.projectName}</ProjectName>
                </FlexRow>
                {this.props.selectedCitations.length > 0 && (
                  <SelectedButtonContainer>
                    {this.props.user.loggedIn ? (
                      <Dropdown
                        name="Copy Selected"
                        items={this.exportSelectedMenu}
                        params={this.getParams()}
                        newButton
                      />
                    ) : (
                      <SelectedButton primary onClick={() => this.copyAll(true)}>
                        Copy Selected
                      </SelectedButton>
                    )}
                    <SelectedButton primary onClick={this.deleteSelected}>
                      Delete Selected
                    </SelectedButton>
                  </SelectedButtonContainer>
                )}
                {this.props.citations.length > 1 && (
                  <SortContainer className="header">
                    <Dropdown
                      name={sortMenu.name}
                      onSelect={key => {
                        this.props.handleSortChange(this.props.citations, key)
                      }}
                      items={sortMenu.menuItems}
                    />
                  </SortContainer>
                )}
              </FlexRow>
              {this.props.citations.length > 0 && (
                <SelectContainer>
                  <CheckboxContainer>
                    <input
                      type="checkbox"
                      id="checkbox"
                      onChange={this.handleCheck}
                      checked={this.state.checked}
                    />
                    {this.props.selectedCitations.length ? (
                      <OrangeLabel htmlFor="checkbox">
                        {this.props.selectedCitations.length} Selected
                      </OrangeLabel>
                    ) : (
                      <Label htmlFor="checkbox"> Select All</Label>
                    )}
                  </CheckboxContainer>
                </SelectContainer>
              )}
            </div>
          </div>
        </div>
      </span>
    )
  }
}
 
const mapStateToProps = state => ({
  header: 'Citation List',
  citations: state.citations,
  user: state.user,
  projectName: state.currentProject.name,
  projectId: state.currentProject.id,
  selected: getCitationStyle(state),
  showSave: state.user.role === 'not_logged_in',
  selectedCitations: getSelectedCitations(state),
})
 
const mapDispatchToProps = dispatch => ({
  handleFetchProjects: citationID => {
    dispatch(showCitationModals(citationID))
  },
  handleSortChange: (citations, sortKey) => {
    dispatch(loadCitations(sortCitations(citations, sortKey)))
    dispatch(gtmSortCitationsEvent(sortKey))
  },
  updateCitations: citations => {
    dispatch(loadCitations(citations))
  },
  handleDelete: citation => {
    dispatch(deleteCitation(citation))
  },
})
 
CitationsListTop.propTypes = propTypes
 
export default connect(
  mapStateToProps,
  mapDispatchToProps
)(CitationsListTop)
