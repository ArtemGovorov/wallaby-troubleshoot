import React from 'react'
import PropTypes from 'prop-types'

import { connect } from 'react-redux'
import { deleteCitation } from 'redux-citation-api'
import { getCitationStyle } from 'easybib/reducers'
 
import Advert from 'easybib/atoms/Advert'
import Button from 'easybib/atoms/Button'
import Citation from 'easybib/organisms/Citation'
import { IconKeyboardArrowRight } from 'easybib/atoms/Icons'

import { loadCitation } from 'easybib/actions/actions.search'

import { getFootnote, showCitationModals } from 'easybib/actions/actions.bibliography'

import { getCitation } from 'easybib/actions/actions.form'

import { showModal } from 'easybib/actions/actions.modals'

import {
  redirectToHomePage,
  nonProRedirect,
  redirectToEditForm,
} from 'easybib/actions/actions.redirects'

const propTypes = {
  citations: PropTypes.arrayOf(
    PropTypes.shape({
      text: PropTypes.string,
      type: PropTypes.string,
    })
  ),
  startCreatingCitation: PropTypes.func,
  loaded: PropTypes.bool,
  emptyCitations: PropTypes.bool,
}

const adPlacements = [
  { placement: 'C2', range: { start: 0, end: 1 } },
  { placement: 'C3', range: { start: 1, end: 4 } },
  { placement: 'C4', range: { start: 4, end: 7 } },
  { placement: 'C5', range: { start: 7 } },
]
 
const CitationsListItems = props => (
  <div className="citations center-children">
    <div>
      <div className="citations-list center-children">
        {props.loaded && props.emptyCitations ? (
          <React.Fragment>
            <div className="no-citations">
              <p>You don&apos;t have any citations yet.</p>
              <Button
                onePress
                theme="secondary"
                size="lg"
                onClick={props.startCreatingCitation}
              >
                Start creating citations <IconKeyboardArrowRight />
              </Button>
            </div>
          </React.Fragment>
        ) : (
          adPlacements.map(item => (
            <React.Fragment key={item.placement}>
              <Advert placement={item.placement} />
              {props.citations.slice(item.range.start, item.range.end).map(citation => (
                <Citation
                  citation={citation}
                  key={citation.id}
                  projectId={props.projectId}
                  citationStyle={props.citationStyle}
                  handleDelete={props.handleDelete}
                  handleEditRedirect={props.handleEditRedirect}
                  handleInText={props.handleShowModal}
                  user={props.user}
                  handleFetchProjects={props.handleFetchProjects}
                />
              ))}
            </React.Fragment>
          ))
        )}
      </div>
    </div>
  </div>
)
 
const mapStateToProps = state => ({
  emptyCitations: state.citations.length === 0,
  loaded: state.app.bibliographyLoaded,
  header: 'Citation List',
  citations: state.citations,
  user: state.user,
  projectId: state.currentProject.id,
  citationStyle: getCitationStyle(state),
})
 
const mapDispatchToProps = dispatch => ({
  handleFetchProjects: citationID => {
    dispatch(showCitationModals(citationID))
  },
 
  handleDelete: citation => {
    dispatch(deleteCitation(citation))
  },

  handleEditRedirect: async (projectId, citationId, sourceType) => {
    await dispatch(getCitation(projectId, citationId))
    dispatch(redirectToEditForm(projectId, citationId, sourceType))
  },
 
  handleShowModal: (citation, style) => {
    dispatch(nonProRedirect()).then(() => {
      dispatch(loadCitation(citation))
      if (style === 'chicago' || style === 'turabian') {
        dispatch(getFootnote(citation, style))
        dispatch(showModal('footnote'))
      } else {
        dispatch(showModal('intext-citation'))
      }
    })
  },

  startCreatingCitation: () => {
    dispatch(redirectToHomePage())
  },
})
 
CitationsListItems.propTypes = propTypes

export default connect(
  mapStateToProps,
  mapDispatchToProps
)(CitationsListItems)
