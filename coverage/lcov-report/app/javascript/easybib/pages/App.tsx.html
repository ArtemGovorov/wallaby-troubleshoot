import React, { Component } from 'react'
import { Route, Switch } from 'react-router'

import { ThemeProvider } from 'styled-components'
import { easybibTheme } from 'easybib/theme'

import { Provider } from 'react-redux'
import store from 'easybib/store'
 
import initGTM from 'easybib/utils/initGTM'
import initHotjar from 'easybib/utils/initHotjar'
import isWindowDefined from 'easybib/utils/isWindowDefined'
import {
  init as wtaInit,
  AnalyticsClientConfig,
} from 'easybib/utils/analyticsFrontendAdapter'
 
import Advert from 'easybib/atoms/Advert'
import AppHeader from 'easybib/components/AppHeader'
import Footer from 'easybib/molecules/Footer'
 
import Home from 'easybib/pages/Home'
import Style from 'easybib/pages/Style'
import ResultsConnected from 'easybib/pages/Results'
import EvalConnected from 'easybib/pages/Eval'
import FormConnected from 'easybib/pages/Form'
import BibliographyConnected from 'easybib/pages/Bibliography'
import initSentry from 'easybib/utils/initSentry'
 
import EBRouter from 'easybib/EBRouter'
import { BackendOpts } from 'types/easybib'
import CenterNode from './CenterNode'
 
interface AppProps {
  url: string
  backendOpts?: BackendOpts
}

class App extends Component<AppProps> {
  constructor(props) {
    super(props)
    if (isWindowDefined) {
      const wtaConfig: AnalyticsClientConfig = {}
      wtaInit(wtaConfig)
    }
  }
  componentDidMount() {
    initGTM()
    initSentry()
    initHotjar()
  }

  render() {
    return (
      <EBRouter url={this.props.url} backendOpts={this.props.backendOpts}>
        <ThemeProvider theme={easybibTheme}>
          <Provider store={store}>
            <React.Fragment>
              <AppHeader withOutRouter={false} />
              <CenterNode>
                <Switch>
                  <Route path="/home" component={Home} />
                  <Route path="/style" component={Style} />
                  <Route path="/style/:style?/:item_type?-citation" component={Style} />
                  <Route
                    path="/:style/:item_type?-citation/search"
                    component={ResultsConnected}
                  />
                  <Route
                    path="/:style/:item_type?-citation/eval"
                    component={EvalConnected}
                  />
                  <Route
                    path="/:style/:item_type?-citation/new"
                    component={FormConnected}
                  />
                  <Route
                    path="/:style-format/:item_type-citation"
                    component={FormConnected}
                  />
                  <Route
                    path="/:style-format/:item_type-citation/pubtab/:tab_id"
                    component={FormConnected}
                  />
                  <Route path="/cite/form" component={FormConnected} />
                  <Route path="/project" component={BibliographyConnected} />
                  <Route path="/project/style/:style" component={BibliographyConnected} />
                  <Route component={Home} />
                </Switch>
              </CenterNode>
              <div id="easybib-bottom-add" />
              <footer id="easybib_footer">
                <Footer />
                <Advert placement="D_SKIN" />
                <Advert placement="D_SB" />
                <Advert placement="D_MISC" />
              </footer>
            </React.Fragment>
          </Provider>
        </ThemeProvider>
      </EBRouter>
    )
  }
}

export default App
