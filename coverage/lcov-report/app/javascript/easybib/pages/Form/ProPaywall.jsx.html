import React from 'react'
import * as constants from 'easybib/constants'
import styled from 'styled-components'
import { connect } from 'react-redux'
import PropTypes from 'prop-types'
 
const WarningContainer = styled.div`
  border: 2px solid ${props => props.theme.colors.yellow};
  background: ${props => props.theme.colors.yellowLight};
  margin-top: ${props => props.theme.spacing.md};
  padding: ${props => props.theme.spacing.md};
`
const ProLink = styled.a`
  color: ${props => props.theme.colors.blue};
  &:hover,
  &:focus {
    color: ${props => props.theme.colors.blue};
  }
`
 
export const ProPaywall = props =>
  props.showProPaywall && (
    <WarningContainer>
      <p>We&apos;re sorry!</p>
      <p>
        Premium styles like APA & Chicago are only available in EasyBib Plus. You can keep
        citing in MLA, or you can sign up for{' '}
        <ProLink href={constants.PRICING_PATH}>a free trial to EasyBib Plus!</ProLink>
      </p>
    </WarningContainer>
  )
 
ProPaywall.propTypes = {
  showProPaywall: PropTypes.bool,
}
 
const mapStateToProps = state => ({
  showProPaywall: state.showProPaywall,
})
 
export default connect(mapStateToProps)(ProPaywall)
