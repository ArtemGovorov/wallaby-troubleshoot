/**
 * Add star icon to recommended fields when empty
 * @see https://github.com/easybib/sbm-issues/issues/1905
 */
function addStarIcons() {
  var emptyClass = 'recommendedHighlightEmpty'

  $('.recommended').each(function() {
    $(getStarIcon()).insertBefore($(this))
    if (
      typeof $(this).val() !== undefined &&
      (!$(this).val() ||
        $(this)
          .val()
          .trim() === '')
    ) {
      $(this).addClass(emptyClass)
    }
  })
 
  $('svg.icon-grade').each(function() {
    // Some inputs have no <span> around <svg><input> so add one to be able to
    // correctly position the star icon.
    var $parent = $(this).parent()
    var parentNode = $parent[0].localName
 
    if (parentNode === 'p') return true // top-desc
    if (parentNode === 'span') {
      if (
        $(this)
          .next('input')
          .hasClass(emptyClass)
      )
        $parent.addClass(emptyClass)
      return true
    }
    var spanClass = 'input-wrapper'
    if (
      $(this)
        .next('input')
        .hasClass(emptyClass)
    )
      spanClass += ' ' + emptyClass
    var span = '<span class="' + spanClass + '"></span>'
    $parent.wrapInner($(span))
  })
}

/**
 * On website/newspaper/magazine forms, the first select redirects to another form
 */
function bindSourceTypeSelect() {
  /**
   * Construct the new form URL based on citation type and selected tab
   * @param {string} value - citation type
   * @param {string} tab
   * @return {string}
   */
  function getFormUrl(value, tab) {
    var origin = window.location.origin
    var pathname = window.location.pathname
    var route = pathname.split('/', 3)[1]
    return origin + '/' + route + '/' + value + '-citation/pubtab/' + tab
  }

  $('select.redirect_pulldown').on('change', function() {
    var tab = $(this)
      .attr('id')
      .replace('pubtab_', '')
    var type = this.value
    /**
     * Special case for newspaper and magazine which have values like:
     * - editorial_pubmagazine_pubonline
     * - editorial_pubnewspaper_pubonline
     * - review_pubnewspaper_pubonline
     * - review_pubmagazine_pubonline
     */
    if (this.value.includes('_pubonline')) {
      var split = this.value.split('_')
      type = split[0]
      tab = split[1]
    }
    window.location.href = getFormUrl(type, tab)
  })
}

/**
 * <input><p><span><label> -> <label /><span><input>
 */
var uniqueIds = 666
function inputId($formElement) {
  // gets the form elements id. Creates one if it's not there.
  var form_el_id = $formElement.attr('id')
  if (!form_el_id) {
    form_el_id = 'form_el_' + uniqueIds
    uniqueIds++
    $formElement.attr('id', form_el_id)
  }
  return form_el_id
}
function fixFormLabel($form) {
  $form.find('td').each(function() {
    if ($(this).has('p.sub_label_bottom').length === 0) return
 
    var $formElement = $(this).children('input')
    if ($formElement.length === 0) $formElement = $(this).children('select')
    var $label = $(this).children('p.sub_label_bottom')

    if ($label.length === 0) return

    var text = $label.children('span').html()
    var formElementId = inputId($formElement)
 
    var $newLabel = $('<label for="' + formElementId + '">' + text + '</label>')
    var $wrapper = $('<span class="input-wrapper"></span>')
    $formElement.before($newLabel)
    $formElement.wrap($wrapper)
    $label.remove()
  })
  // fix up the inputs to have unique ids.
  $form.find('input').each(function() {
    if (!$(this).attr('id') && $(this).attr('name')) {
      $(this).attr('id', $(this).attr('name'))
    }
  })
  // fix checkboxes to be wrapped in a label, and text in a span.
  $form.find('input[type="checkbox"]').each(function(index, el) {
    $(el)
      .parent()
      .wrapInner('<label></label>')
    $(el)
      .parent()
      .each(function(index, v) {
        $(v)
          .contents()
          .eq(1)
          .wrap('<span class="label-span" />')
      })
  })
}

// formDataToJson is from the legacy js.
function getSection(formfield) {
  return formfield.substring(0, formfield.indexOf('['))
}
function getColumn(formfield) {
  return formfield.substring(formfield.indexOf('[') + 1, formfield.indexOf(']'))
}

function formDataToJson() {
  var formData = {}
  formData.source = $('#source').val()

  if (typeof formData.source == 'undefined') {
    formData.source = $('input[name=source]').val()
  }
 
  formData.pubtype = { main: $('#pubtype').val(), suffix: $('#pubsuffix').val() }

  var fields = $('#citation_form').find(
    'input:radio:checked, input:text, select, input:checkbox:checked'
  )
  var other = $('.section_publication input[type=hidden]:not("#pubtype, #pubsuffix")')
 
  fields = fields.add(other)
  $(fields).each(function() {
    var fieldName = $(this).attr('name')
    if (typeof fieldName != 'undefined' && fieldName.length > 0) {
      var valueName = $(this).attr('name')
      var valueSection = getSection(valueName)
      var valueColumn = getColumn(valueName)
      if (typeof formData[valueSection] == 'undefined') {
        formData[valueSection] = {}
      }
      formData[valueSection][valueColumn] = $(this).val()
    }
  })
  formData.contributors = []
  $('#citation_form .contributors_table_row').each(function(i) {
    var contribRow = {}
    contribRow['function'] = $(this)
      .find('.contributors_function')
      .val()
    contribRow['first'] = $(this)
      .find('.contributors_first')
      .val()
    contribRow['middle'] = $(this)
      .find('.contributors_middle')
      .val()
    contribRow['last'] = $(this)
      .find('.contributors_last')
      .val()
    contribRow['suffix'] = $(this)
      .find('.contributors_suffix')
      .val()
    formData.contributors.push(contribRow)
  })
  if ($('#custom-citation') && $('#custom-citation').val()) {
    formData.custom = { citation: $('#custom-citation').val() }
  }

  return formData
}

  function inject(dataArray) {
    var field_name
 
    var otherStuff = ''
 
    for (var section in dataArray) {
      if (section == 'pubtype') {
        selectPubtype(dataArray[section])
      } else if (section == 'contributors') {
        fillContrib(dataArray[section])
      } else if (section == 'other') {
        otherStuff = dataArray[section]
        if (typeof dataArray[section] == 'object') {
          otherStuff = JSON.stringify(dataArray[section])
        }
        $('#other').val(escape(otherStuff))
      } else {
        for (var field in dataArray[section]) {
          if (field == 'month' || field == 'monthaccessed') {
            dataArray[section][field] = dataArray[section][field].toLowerCase()
          }
          field_name = section + '[' + field + ''
          input_name = "[name^='" + field_name + "]']"
          fillValue(input_name, dataArray[section][field])
        }
      }
    }
 
    if (
      typeof dataArray['annotation'] != 'undefined' &&
      typeof dataArray['annotation']['text'] != 'undefined'
    ) {
      if (dataArray['annotation']['text'].length > 0) {
        $('#annotation')
          .find('textarea')
          .val(dataArray['annotation']['text'])
        $('#annotation').show()
      }
    }
  }
 
  /**
   * fillValue
   * Fills out all values outside of contributors.
   * @param {Object} data form data
   */
  function fillValue(field, data) {
    if ($(field).is('textarea')) {
      $(field).html(data)
    }
    var type = $(field).attr('type')
    switch (type) {
      case 'text':
        $(field).val(data)
        break
      case 'radio':
        $(field).each(function(n) {
          if ($(this).val() == data) {
            $(this).attr('checked', 'checked')
            $(this).click()
          }
        })
        break
      case 'checkbox':
        if (data.length > 0) {
          $(field).click()
        }
        break
      default:
        $(field).val(data)
    }
  }
 
  function selectPubtype(data) {
    if (typeof pubtab !== 'undefined' && pubtab.length > 0) {
      return
    }
 
    //var pubtab = '';
    if (data.length == 0) {

    }
    if (data['main']) {
      pubtab = 'main_' + data['main']
    }
    if (pubtab.length > 0 && data['suffix']) {
      pubtab += '_suffix_' + data['suffix']
    } else if (data['suffix']) {
      pubtab = 'suffix_' + data['suffix']
    }
    var s = $('.pubsuffixheader').filter('#' + pubtab)
    var sl = s.length
    if (sl > 0) {
      s.click()
      var p = s
        .parent()
        .children('input')
        .eq(0)
        .attr('id')
      $('.tabheader')
        .filter('#' + p)
        .click()
    } else {
      $('.tabheader')
        .filter('#' + pubtab)
        .click()
    }
  }
 
  /**
   * fillContrib
   * Fills out the contributors upon edit.
   * @param {Object} data contributor data
   */
  function fillContrib(data) {
    $('.contributors_function').val(
      $('.contributors_function')

        .val()
    )
    $('.contributors_first').val('')
    $('.contributors_middle').val('')
    $('.contributors_last').val('')

    var length = data.length
 
    for (var k = 0; k < length; k++) {
      if (k < length - 1) {
        $('a.contrib_add').click()
      }
      var orig_function = $('.contributors_function')
        .eq(k)
        .val()
      $('.contributors_function')
        .eq(k)
        .val(data[k]['function'])
      if (
        $('.contributors_function')
          .eq(k)
          .find(':selected')
          .text().length == 0
      ) {
        $('.contributors_function')
          .eq(k)
          .val(orig_function)
      }
      $('.contributors_first')
        .eq(k)
        .val(data[k]['first'])
      $('.contributors_middle')
        .eq(k)
        .val(data[k]['middle'])
      $('.contributors_last')
        .eq(k)
        .val(data[k]['last'])
      $('.contributors_suffix')
        .eq(k)
        .val(data[k]['suffix'])
    }
