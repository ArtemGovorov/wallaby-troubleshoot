import React, { Component } from 'react'
import PropTypes from 'prop-types'

import Button from 'easybib/components/Button'
import Input from 'easybib/components/Form/Input'
import styled from 'styled-components'
 
const ActualButton = styled(Button.withComponent('button'))`
  margin-left: 2px;
`
 
class SearchSourceSmall extends Component {
  state = { text: '' }

  handleChange = e => {
    this.setState({ text: e.target.value })
  }

  handleSubmit = e => {
    e.preventDefault()
    if (this.state.text.trim().length !== 0) {
      this.props.onSearch(this.state.text, this.props.type)
    }
  }
 
  render() {
    return (
      <div className="main-search-small">
        <p className="info-text">Search for source here</p>
        <div className="inner">
          <form onSubmit={this.handleSubmit} method="POST">
            <Input placeholder="Search for resource here" onChange={this.handleChange} />
            <ActualButton newPrimary disabled={this.state.text.trim() === ''}>
              Cite
            </ActualButton>
          </form>
        </div>
        <p className="info-text">Or Cite it Manually</p>
      </div>
    )
  }
}

SearchSourceSmall.propTypes = {
  type: PropTypes.string,
  onSearch: PropTypes.func,
}
 
export default SearchSourceSmall
