import React, { Component } from 'react'
 
import { connect } from 'react-redux'
 
import { initResultsPage, unloadSources } from 'easybib/actions/actions.results'
import Advert from 'easybib/atoms/Advert'
import ProgressStepper from 'easybib/organisms/ProgressStepper'
import ErrorContainer from 'easybib/environment/ErrorContainer'
import ProPaywall from 'easybib/pages/Form/ProPaywall'
import withInitActionProvider from 'easybib/hocs/withInitActionProvider'
import AdSpinner from 'easybib/components/AdSpinner'
import SearchResults from 'easybib/pages/Results/SearchResults'
 
import Questions from 'easybib/pages/Results/Questions'
interface Props {
  unloadSources: () => void
}

export class Results extends Component<Props> {
  componentWillUnmount() {
    this.props.unloadSources()
  }
 
  render() {
    return (
      <div>
        <AdSpinner page="results">
          <div className="container-fluid">
            <ProPaywall />
            <div>
              <Advert placement="C1" />
              <div className="row">
                <ProgressStepper step={1} />
              </div>
            </div>
            <Advert placement="C2" />
            <ErrorContainer />
            <SearchResults sliceMin={0} sliceMax={1} />
            <Advert placement="C3" />
            <SearchResults sliceMin={1} sliceMax={4} />
            <Advert placement="C4" />
            <SearchResults sliceMin={4} />
            <Advert placement="C5" />
            <Questions />
          </div>
        </AdSpinner>
      </div>
    )
  }
}
 
const mapDispatchToProps = dispatch => ({
  unloadSources: () => dispatch(unloadSources()),
})
 
const Connected = connect(
  null,
  mapDispatchToProps
)(Results)
 
export default withInitActionProvider(Connected, initResultsPage, 'search')
