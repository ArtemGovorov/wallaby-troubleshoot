import CMSContentBox from 'easybib/components/Styled'

const QuestionUL = styled.ul`
  margin-bottom: 0;
`

const Questions: React.SFC<{}> = () => (
  <div className="cms-content results">
    <div className="container-fluid" id="content-addition">

      <div className="row">
        {[

            items: [

              'Have other credible individuals referenced this source or author?',
              'Book: What have reviews said about it?',
            ],
          },
          {
            header: 'Publisher',
            items: [
              'What do you know about the publisher/sponsor? Are they well-respected?',
