import React from 'react'
import uuidV1 from 'uuid'
import { Link } from 'react-router-dom'
import { connect } from 'react-redux'
 
import { getCitationStyle } from 'easybib/reducers'
 
import { searchJournal, useSearchResult } from 'easybib/actions/actions.results'
import { redirectToForm, redirectToHomePage } from 'easybib/actions/actions.redirects'

import Nav from 'easybib/atoms/Nav'
import SearchResultCard from 'easybib/organisms/SearchResultCard'
import { IconKeyboardArrowRight } from 'easybib/atoms/Icons'
import SearchNoResults from 'easybib/pages/Results/SearchNoResults'
import SearchNoResultsYet from 'easybib/pages/Results/SearchNoResultsYet'

interface Props {
  searchType: string
  searchText: string
  sources: { data: {}; display: {} }[]
  handleCite: (
    source: string,
    searchText: string,
    searchType: string,
    style: string
  ) => void
  style: string
  handleJournalSearch: (type: string) => void
  noResults?: boolean
  handleSearchAgain: () => void
  handleManualCitation: () => void
  sliceMin: number
  sliceMax: number
}

export const SearchResults: React.SFC<Props> = props => {
  const journalNavItems = [
    { name: 'Search results', value: 'journal' },
    { name: 'JSTOR', value: 'jstor' },
    { name: 'ProQuest', value: 'proquest' },
  ]
 
  let theSources
  if (!props.sliceMax) {
    theSources = props.sources.slice(props.sliceMin)
  } else {
    theSources = props.sources.slice(props.sliceMin, props.sliceMax)
  }
 
  return (
    <div>
      {props.sliceMin === 0 && (
        <div className="search-title text-center">
          <div>
            <Link className="home" to="/home">
              Home
            </Link>
            <IconKeyboardArrowRight />
            Results for{' '}
            <span className="source-type">
              {props.searchType ? props.searchType.replace('_', ' ') : ''}
            </span>{' '}
            titled <span className="title-name">{props.searchText}</span>
          </div>
          {props.searchType === 'journal' && (
            <Nav
              items={journalNavItems}
              onChange={props.handleJournalSearch}
              selected="journal"
              css="journal-tabs"
            />
          )}
        </div>
      )}
      <div className="search-results text-center">
        {theSources.map(source => (
          <SearchResultCard
            handleCite={props.handleCite}
            item={source}
            key={uuidV1()}
            searchText={props.searchText}
            searchType={props.searchType}
            style={props.style}
          />
        ))}
        {props.noResults &&
          props.sliceMin === 0 && (
            <div>
              <SearchNoResults
                searchAgain={props.handleSearchAgain}
                manualCitation={props.handleManualCitation}
              />
            </div>
          )}
        {!props.noResults &&
          props.sliceMin === 0 &&
          !props.sources.length && <SearchNoResultsYet />}
      </div>
    </div>
  )
}
 
const mapStateToProps = (state, ownProps) => ({
  searchType: state.search.type,
  searchText: state.search.text,
  sources: state.sources,
  style: getCitationStyle(state),
  noResults: state.app.searchLoaded && state.sources.length === 0,
  sliceMin: ownProps.sliceMin,
  sliceMax: ownProps.sliceMax,
})
 
const mapDispatchToProps = dispatch => ({
  handleCite: (source, searchText, searchType, style) =>
    dispatch(useSearchResult(source, searchText, searchType, style)),
  handleJournalSearch: type => dispatch(searchJournal(type)),
  handleSearchAgain: () => dispatch(redirectToHomePage()),
  handleManualCitation: () => dispatch(redirectToForm('')),
})
 
export default connect(
  mapStateToProps,
  mapDispatchToProps
)(SearchResults)
