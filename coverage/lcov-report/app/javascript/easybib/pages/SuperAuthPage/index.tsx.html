import React, { Component } from 'react'
import { connect } from 'react-redux'
import styled from 'styled-components'
 
import isWindowDefined from 'easybib/utils/isWindowDefined'
import { getUserBibInfo } from 'easybib/utils/localStorage'
import { loadCheggSuperAuth } from 'easybib/utils/cheggauth'
import initGTM from 'easybib/utils/initGTM'
 
import ModalWrapper from 'easybib/organisms/ModalWrapper'
import withReduxProvider from 'easybib/hocs/withReduxProvider'

import { initUser } from 'easybib/actions/actions.shared'
import { showModal } from 'easybib/actions/actions.modals'
 
const Container = styled.div`
  padding: 12px;
  margin-bottom: 16px;
  background: white;
`
const EasybibLogo = styled.img.attrs({
  src: '/v2/images/easybib-chegg-logo.svg',
  alt: 'EasyBib',
})`
  width: 111px;
`
 
interface Props {
  showSubModal: (options: {}) => void
  redirectTo: string
  showValueprop?: boolean
  defaultflow: string
  initUser: () => void
  pageName?: string
  ocVersion: string
  client: any
  plusTest?: boolean
  response_type: string
}

export class SuperAuthPage extends Component<Props> {
  async componentDidMount() {
    initGTM()
 
    await this.props.initUser()

    const {
      showSubModal,
      defaultflow,
      redirectTo,
      showValueprop,
      client,
      pageName,
      plusTest,
      ocVersion,
      // eslint-disable-next-line camelcase
      response_type,
    } = this.props
    const { loggedIn, user } = getUserBibInfo()
    if (loggedIn && !user.paidMember) {
      showSubModal({ plusTest })
    } else if (loggedIn && user.paidMember) {
      window.location.href = '/myaccount'
    } else {
      let accurateRedirectTo = redirectTo
      if (
        (pageName === 'upgrade' && !loggedIn) ||
        (pageName === 'register' && plusTest)
      ) {
        accurateRedirectTo = '/upgrade'
      }
      loadCheggSuperAuth({
        // eslint-disable-next-line camelcase
        superAuth: { defaultflow, response_type },
        options: {
          showSubModal,
          showValueprop,
          client,
          ocVersion,
          redirectTo: accurateRedirectTo,
        },
      })
    }
  }
 
  render() {
    return (
      <Container>
        <EasybibLogo />
        {isWindowDefined && <ModalWrapper />}
      </Container>
    )
  }
}
 
const mapDispatchToProps = dispatch => ({
  showSubModal: options => dispatch(showModal('subscription', options)),
  initUser: async () => {
    await dispatch(initUser())
  },
})
 
export default withReduxProvider(
  connect(
    null,
    mapDispatchToProps
  )(SuperAuthPage)
)
