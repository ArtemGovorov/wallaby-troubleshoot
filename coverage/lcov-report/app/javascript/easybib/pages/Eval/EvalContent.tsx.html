import React from 'react'
import { connect } from 'react-redux'
import EvalBook from 'easybib/molecules/EvalBook'
import EvalJournal from 'easybib/molecules/EvalJournal'
import EvalWebsite from 'easybib/molecules/EvalWebsite'
import { Citation } from 'types/easybib'
 
interface Props {
  citation: Citation
}
 
const sourceToEvalComponent = {
  book: EvalBook,
  journal: EvalJournal,
}

export const EvalContent: React.SFC<Props> = ({ citation }) => {
  const { source } = citation;
  if (!source) return null
  const EvalPage = sourceToEvalComponent[source] || EvalWebsite
  return <EvalPage citation={citation} />
}
 
const mapStateToProps = state => ({
  citation: state.citation,
})

export default connect(mapStateToProps)(EvalContent)
