import React, { Component } from 'react'
import PropTypes from 'prop-types'
import Infinite from 'react-infinite'
import { connect } from 'react-redux'
import onClickOutside from 'react-onclickoutside'
import { setCitationStyle } from 'easybib/actions/actions.shared'
import {
  nonProRedirect,
  redirectToBibliographyPage,
} from 'easybib/actions/actions.redirects'
import {
  loadProjectAndCitations,
  loadMoreStyles,
} from 'easybib/actions/actions.bibliography'
import { POPULAR_STYLES } from 'easybib/constants'
import Icon from 'easybib/components/Icon'
import { getCitationStyle } from 'easybib/reducers'
import SearchBox from '../atoms/SearchBox'
import isWindowDefined from '../utils/isWindowDefined'
import { bibStyleEvent, styleEvent } from '../utils/gtm'
 
const UP = 38
const DOWN = 40
const ENTER = 13
const ESCAPE = 27

class StylesDropdown extends Component {
  constructor(props) {
    super(props)
    this.state = {
      open: false,
      // start with page=2 because in actions::initHomePage() we already call page=1
      page: 2,
      searchTerm: '',
      focused: false,
      highlighted: '',
      isLoading: false,
      searchError: false,
    }
    this.handleFocus = this.handleFocus.bind(this)
    this.handleInfiniteLoad = this.handleInfiniteLoad.bind(this)
    this.handleKeyUp = this.handleKeyUp.bind(this)
    this.handleKeyDown = this.handleKeyDown.bind(this)
    this.handleSearch = this.handleSearch.bind(this)
    this.handleBlur = this.handleBlur.bind(this)
  }
 
  componentDidMount() {
    window.addEventListener('keyup', this.handleKeyUp)
    window.addEventListener('keydown', this.handleKeyDown)
  }

  componentWillUnmount() {
    window.removeEventListener('keyup', this.handleKeyUp)
    window.removeEventListener('keydown', this.handleKeyDown)
  }

  handleFocus() {
    this.setState({ focused: true })
  }

  handleBlur() {
    // setTimeout used here to work around event priority issue in chrome.
    //   which prevents onclick being applied to a button somehow.
    setTimeout(() => this.setState({ focused: false }), 10)
  }

  handleClickOutside() {
    this.setState({ open: false, isLoading: false })
  }
 
  handleKeyDown(event) {
    if (this.state.open || this.state.focused) {
      if ([UP, DOWN].includes(event.keyCode)) {
        event.preventDefault()
      }
    }
  }
 
  handleKeyUp(event) {
    const key = event.keyCode || event.charCode || 0
    if (this.state.open) {
      if (key === UP) {
        this.moveHighlighted(false)
        event.preventDefault()
      } else if (key === DOWN) {
        this.moveHighlighted(true)
        event.preventDefault()
      } else if (key === ESCAPE) {
        this.setState({ highlighted: '' })
        this.toggleOpen()
        event.preventDefault()
      } else if (key === ENTER) {
        this.selectHighlighted()
        this.setState({ open: false })
        event.preventDefault()
      }
    } else if (!this.state.open && this.state.focused) {
      if ([UP, DOWN].includes(key)) {
        this.toggleOpen()
        event.preventDefault()
      }
    }
  }
 
  highlightedIndex() {
    const found = this.props.styles.find(
      option => this.state.highlighted === (option.alias || option.value || option.code)
    )
    return found ? this.props.styles.indexOf(found) : null
  }
 
  scrollToHighlighted(index) {
    this.itemsListRef.querySelector('div').scrollTop = 50 * index
  }

  moveHighlighted(next) {
    // move the highlighted item up or down.
    let index = this.highlightedIndex()
    if (index !== null) {
      if (next) {
        const inBounds = index + 1 > this.props.styles.length - 1
        index = inBounds ? this.props.styles.length - 1 : index + 1
      } else {
        // previous item.
        index = index - 1 < 0 ? 0 : index - 1
      }
      const option = this.props.styles[index]
      this.setState({ highlighted: option.alias || option.value || option.code })
      this.scrollToHighlighted(index)
    }
  }

  selectHighlighted() {
    const index = this.highlightedIndex()
    if (index !== null) this.handleChange(this.props.styles[index])
  }
 
  toggleOpen() {
    const opening = !this.state.open // state.open does not yet reflect the "open" status
    if (this.props.disabled) return
    if (!opening) this.setState({ searchError: false })

    this.setState({
      open: !this.state.open,
      highlighted: this.props.selected,
    })
  }

  handleChange(option) {
    const optionValue = option.alias || option.value || option.code // bmcm uses `option.code` for styles
    this.setState({ open: false })
    if (optionValue !== this.props.selected) {
      this.props.handleChange(optionValue)
    }
  }

  handleSearch(term) {
    const page = 1
    const merge = false
    this.setState({ searchTerm: term, page, isLoading: true })
    this.props.loadMore(term, page, merge, results =>
      this.setState({ searchError: !results.length, isLoading: false })
    )
  }

  /**
   * Reset styles
   */
  handleCancelSearch() {
    const searchTerm = ''
    const page = 1
    const merge = false
    if (isWindowDefined) {
      document.body.style.overflow = 'inherit'
    }
    this.setState({ searchTerm, page, searchError: false, isLoading: false })
    if (this.props.hasSearch) this.props.loadMore(searchTerm, page, merge)
  }
 
  handleInfiniteLoad() {
    if (!this.props.loadMore) return
    if (!this.state.open) return
    this.setState({ isLoading: true })
    const merge = true
    this.props.loadMore(this.state.searchTerm, this.state.page, merge, () => {
      this.setState({ isLoading: false })
    })
    this.setState({ page: this.state.page + 1 })
  }

  render() {
    let cssClass = 'dropdown hidden-xs'
    const fixName = name => (name === 'MLA' ? 'MLA7' : name)
    const liOptionClassName = option => {
      let className = option.id === this.props.selected ? 'disabled' : ''
      if (this.state.highlighted === (option.alias || option.value || option.code))
        className += ' highlighted'
      return className
    }

    const selectedOption = this.props.styles.find(
      option => this.props.selected === (option.alias || option.value || option.code)
    )
 
    if (isWindowDefined && !this.state.open) document.body.style.overflow = 'inherit'
 
    if (selectedOption) {
      cssClass += ' is-selected'
    }

    const currentOption = selectedOption ? fixName(selectedOption.name) : 'More'
 
    if (this.props.className) cssClass += ` ${this.props.className}`
    if (this.props.disabled) {
      cssClass = `${cssClass} is-disabled`
    } else if (this.state.open) cssClass = `${cssClass} is-open`

    const ulClass = 'items ignore-react-onclickoutside'
    const baseItemClass = 'dropdown-item'
 
    return (
      <div>
        <span className="input-wrapper">
          <select
            onChange={e => this.props.handleChange(e.target.value)}
            className="visible-xs"
            value={this.props.selected}
          >
            {POPULAR_STYLES.map(option => (
              <option value={option.value} key={option.value}>
                {option.name}
              </option>
            ))}
          </select>
        </span>
        <div className={cssClass} tabIndex="0">
          {this.props.label && <label htmlFor="dd-button">{this.props.label}</label>}
          <button
            id="dd-button"
            type="button"
            className="click-wrapper"
            onClick={() => this.toggleOpen()}
            onFocus={this.handleFocus}
            onBlur={this.handleBlur}
            aria-expanded={this.state.open}
          >
            <Icon
              name="downArrow"
              className="pull-right down-arrow-icon material-icons"
            />
            <span className="title">{currentOption}</span>
          </button>
          <div
            className="citation-list-wrapper"
            aria-expanded={this.state.open}
            onMouseEnter={() => {
              // to prevent the window scrolling when we are scrolling the dropdown.
              document.body.style.overflow = 'hidden'
            }}
            onMouseLeave={() => {
              document.body.style.overflow = 'inherit'
            }}
          >
            {this.state.isLoading && (
              <div className="spinner">
                <i className="spin-icon" />
              </div>
            )}
            <ul
              className={ulClass}
              role="listbox"
              ref={ul => {
                this.itemsListRef = ul
              }}
            >
              <li className={`${baseItemClass} searchbox`}>
                <SearchBox
                  placeholder="Search"
                  handleSearch={term => this.handleSearch(term)}
                  cancelSearch={() => this.handleCancelSearch()}
                  searching={this.props.searching}
                  reset={this.state.open}
                />
              </li>
              {this.state.searchError && (
                <li className={`${baseItemClass} search-error-text`}>
                  No matching styles found
                </li>
              )}
              {!this.state.searchError && (
                <Infinite
                  containerHeight={this.props.containerHeight}
                  elementHeight={50}
                  infiniteLoadBeginEdgeOffset={20}
                  onInfiniteLoad={this.handleInfiniteLoad}
                >
                  {this.props.styles.map(option => (
                    <li key={option.id} className={liOptionClassName(option)}>
                      <button
                        onClick={() => {
                          this.handleChange(option)
                        }}
                        role="option"
                        type="button"
                        aria-selected={option.name === this.props.selected}
                      >
                        {fixName(option.name)}
                        {this.props.selected ===
                          (option.value || option.alias || option.code) && (
                          <Icon
                            name="checkmark"
                            className="pull-right check-icon material-icons"
                          />
                        )}
                      </button>
                    </li>
                  ))}
                </Infinite>
              )}
            </ul>
          </div>
        </div>
      </div>
    )
  }
}

StylesDropdown.propTypes = {
  styles: PropTypes.arrayOf(
    PropTypes.shape({
      id: PropTypes.oneOfType([PropTypes.number, PropTypes.string]),
      name: PropTypes.string,
      value: PropTypes.string,
    })
  ).isRequired,
  label: PropTypes.string,
  selected: PropTypes.string,
  disabled: PropTypes.bool,
  hasSearch: PropTypes.bool,
  loadMore: PropTypes.func,
  searching: PropTypes.bool,
  containerHeight: PropTypes.number,
}

StylesDropdown.defaultProps = {
  containerHeight: 300,
}
 
const mapStateToProps = state => ({
  styles: state.styles || [],
  selected: getCitationStyle(state),
})
 
const mapDispatchToProps = (dispatch, ownProps) => ({
  loadMore: (term, page, merge, callback) => {
    dispatch(loadMoreStyles(term, page, merge, callback))
  },
  handleChange: style => {
    if (ownProps.page === 'project') {
      bibStyleEvent(style)
    } else {
      styleEvent(style)
    }
 
    const setStyleThenRedirect = () => {
      dispatch(setCitationStyle(style))
      if (ownProps.page === 'project') {
        dispatch(redirectToBibliographyPage({ style }))
        if (window.gon.cell.spa) dispatch(loadProjectAndCitations())
      }
    }

    if (style !== 'mla7' && style !== 'mla8') {
      dispatch(nonProRedirect()).then(setStyleThenRedirect)
    } else {
      setStyleThenRedirect()
    }
  },
})
 
StylesDropdown.propTypes = {
  handleChange: PropTypes.func,
  className: PropTypes.string,
  styles: PropTypes.arrayOf(
    PropTypes.shape({
      name: PropTypes.string,
      value: PropTypes.string,
    })
  ).isRequired,
}
 
export default connect(
  mapStateToProps,
  mapDispatchToProps
)(onClickOutside(StylesDropdown))
