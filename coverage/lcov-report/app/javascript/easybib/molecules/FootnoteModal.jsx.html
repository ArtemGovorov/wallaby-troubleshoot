import React, { Component } from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
import clipboard from '../utils/copypaste'
import Button from '../atoms/Button'
import Input from '../atoms/Input'
import { updateCitation } from '../actions/actions.form'

/* eslint-disable react/no-danger */
class FootnoteModal extends Component {
  constructor(props) {
    super(props)
    this.state = { input: '', copied: false }
    this.handleFootnote = this.handleFootnote.bind(this)
    this.handleCopy = this.handleCopy.bind(this)
    this.handleInput = this.handleInput.bind(this)
  }

  handleCopy() {
    const html = this.handleFootnote()
    clipboard.copy({
      'text/plain': html.replace(/(<([^>]+)>)/gi, ''),
      'text/html': html,
    })
    this.setState({ copied: true })
  }
 
  handleFootnote() {
    const replace = '|PAGE|'
    const text = this.props.citation.footnote || ''
    return text.replace(replace, this.state.input)
  }
 
  handleInput(e) {
    this.setState({ input: e.target.value })
  }
 
  render() {
    return (
      <div>
        <div className="row">
          <div className="col-sm-12">
            <p className="description">
              Type in the page number you are citing in the box below. The citation will
              update based on your page.
            </p>
          </div>
        </div>
        <div className="row padded flex-center">
          <div
            className="col-sm-8"
            dangerouslySetInnerHTML={{ __html: this.handleFootnote() }}
          />
          <div className="col-sm-4">
            <Button theme="primary" size="lg" onClick={this.handleCopy}>
              {this.state.copied ? 'Copied' : 'Copy'}
            </Button>
          </div>
        </div>
        <div className="row padded-top">
          <div className="col-sm-12">
            <Input
              placeholder="e.g., 'Summary'"
              label="Area where citation appears"
              onChange={this.handleInput}
              value={this.state.input}
            />
          </div>
        </div>
      </div>
    )
  }
}
/* eslint-enable */
 
const mapStateToProps = state => ({
  citation: state.citation,
})

const mapDispatchToProps = dispatch => ({
  handleSaveCitation: citation => {
    const link = citation.links.me
    dispatch(updateCitation(link, citation))
  },
})

FootnoteModal.propTypes = {
  citation: PropTypes.shape({
    footnote: PropTypes.string,
  }),
}

export default connect(mapStateToProps, mapDispatchToProps)(FootnoteModal)
