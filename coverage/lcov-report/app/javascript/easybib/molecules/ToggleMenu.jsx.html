import React, { Component } from 'react'
import PropTypes from 'prop-types'
import styled from 'styled-components'
import { MaterialIcon, ActionButton } from 'easybib/organisms/Citation/styles'
import ClickOutside from './ClickOutside'

const DropdownToggle = styled(ActionButton)`
  &:focus {
    outline: 0;
  }
`

const CoordsOrigin = styled.span`
  position: relative;
`

class ToggleMenu extends Component {
  constructor(props) {
    super(props)
    this.state = { open: false }
  }

  handleToggle = () => {
    this.setState({ open: !this.state.open })
    this.props.onClick()
  }
 
  render() {
    const { children, onClick, ...otherProps } = this.props
    return (
      <CoordsOrigin>
        <DropdownToggle
          {...otherProps}
          type="button"
          className="dropdown-toggle"
          aria-haspopup
          onClick={this.handleToggle}
        >
          <MaterialIcon name="more horizontal" />
        </DropdownToggle>
        <ClickOutside onClickOutside={this.handleToggle}>
          {this.state.open && <span onClick={this.handleToggle}>{children}</span>}
        </ClickOutside>
      </CoordsOrigin>
    )
  }
}
 
ToggleMenu.propTypes = {
  children: PropTypes.node,
  onClick: PropTypes.func,
}

export default ToggleMenu
