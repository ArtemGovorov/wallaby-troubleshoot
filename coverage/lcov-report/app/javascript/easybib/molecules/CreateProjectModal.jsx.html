import React, { Component } from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
 
import { getCitationStyle } from 'easybib/reducers'

import Button from '../atoms/Button'
import { createNamedProject } from '../actions/actions.bibliography'
 
import { hideModal } from '../actions/actions.modals'
 
class CreateProjectModal extends Component {
  constructor() {
    super()
    this.state = { input: '' }
    this.handleSubmit = this.handleSubmit.bind(this)
  }

  handleChange(e) {
    this.setState({ input: e.target.value })
  }

  handleSubmit() {
    if (this.state.input !== '') this.props.createProject(this.state.input)
  }

  render() {
    return (
      <div>
        <div className="text-center" style={{ marginBottom: '10px' }}>
          You do not have another project. Enter a project name to start a new one.
        </div>
        <div className="text-center">
          <input
            onChange={e => this.handleChange(e)}
            value={this.state.input}
            className="form-control"
            style={{ marginRight: '15px', marginBottom: '15px' }}
          />
          <div>
            <Button
              theme="primary"
              size="lg"
              style={{ marginRight: '5px' }}
              disabled={!this.state.input}
              onClick={this.handleSubmit}
            >
              Create Project and Copy
            </Button>
            <Button theme="secondary" size="lg" onClick={this.props.hideModal}>
              Cancel
            </Button>
          </div>
        </div>
      </div>
    )
  }
}
 
const mapStateToProps = state => ({
  style: getCitationStyle(state),
  citation: state.citation,
})
 
const mapDispatchToProps = dispatch => ({
  hideModal: () => {
    dispatch(hideModal())
  },
  createProject: name => {
    dispatch(createNamedProject(name))
  },
})

CreateProjectModal.propTypes = {
  hideModal: PropTypes.func,
  createProject: PropTypes.func,
}
 
export default connect(mapStateToProps, mapDispatchToProps)(CreateProjectModal)
