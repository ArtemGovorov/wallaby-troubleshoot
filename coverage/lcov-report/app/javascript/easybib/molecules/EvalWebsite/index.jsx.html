import { trackWithoutThrow } from 'easybib/utils/track'
import Eval from '../Eval'

const ucFirst = string => (string ? string.charAt(0).toUpperCase() + string.slice(1) : '')
 
const formatDate = (month, day, year) => {
  let date = ''
  if (month !== '') date += month
  if (day !== '') {
    if (date.trim() !== '') date += ' '
    date += day
  }
  if (year) {
    if (date.trim() !== '') date += ', '
    date += year
  }
  return date
}
 
const getAccessedDate = data => {
  if (!data) return ''
  const day = data.dayaccessed
  const month = ucFirst(data.monthaccessed)
  const year = data.yearaccessed
 
  return formatDate(month, day, year)
}

const getPublishedDate = data => {
  if (!data) return ''
  const { day } = data
  const month = ucFirst(data.month)
  const { year } = data

  return formatDate(month, day, year)
}

export default class EvalWebsite extends Eval {
  constructor(props) {
    super(props)
    this.state = {
      fetched: this.props.fetched || false, // for testing
      fields: [],
      found: [],
      missing: [],
    }
    this.evaluateAsync = this.evaluateAsync.bind(this)
  }

  /**
   * This will only be evaluated in tests
   */
  componentWillMount() {
    if (this.props.citation) {
      this.setState({ fetched: true })
      this.evaluateAsync(this.props.citation)
    }
  }

  /**
   * this.props.citation will be set asynchronously (autocite) so unlike other Eval* we
   * declare `fields` here so their value is not set to undefined when component is mounted
   * without citation props yet
   * @param {Object} citation
   */
  evaluateAsync(citation) {
    if (!citation || Object.keys(citation).length === 0) return

    const fields = [
      { label: 'Article Title', value: citation.website && citation.website.title },
      { label: 'Publisher / Sponsor', value: citation.other && citation.other.sponsor },
    ]
    if (citation.pubonline) {
      fields.push(
        { label: 'URL', value: citation.pubonline.url },
        { label: 'Website Title', value: citation.pubonline.title },
        { label: 'Date Accessed', value: getAccessedDate(citation.pubonline) },
        { label: 'Date Published', value: getPublishedDate(citation.pubonline) }
      )
    } else trackWithoutThrow(new Error('no pubonline'), null, { citation })
    const found = []
    const missing = []
    fields.forEach(field => {
      if (field && field.value && field.value !== '') {
        found.push(field)
      } else {
        missing.push({ label: field.label, value: 'Edit on next step' })
      }
    })
    this.setState({ found, missing })
  }
}
