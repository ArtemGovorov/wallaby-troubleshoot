import React, { Component } from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
 
import { getCitationStyle } from 'easybib/reducers'

import Button from '../atoms/Button'
import { saveCitationsToProject } from '../actions/actions.bibliography'
import { hideModal, unloadCitation } from '../actions/actions.modals'

class SelectProjectModal extends Component {
  constructor(props) {
    super(props)
    this.state = { selectedProjectID: null }
    this.handleChange = this.handleChange.bind(this)
    this.handleSubmit = this.handleSubmit.bind(this)
  }

  handleChange(event) {
    const index = event.target.selectedIndex
    const optionElement = event.target.childNodes[index]
    this.setState({ selectedProjectID: optionElement.id })
  }

  handleSubmit() {
    this.props.saveCitations(this.state.selectedProjectID)
  }

  render() {
    return (
      <div style={{ margin: '20px 0' }}>
        <center>
          <p>Please select the project you&apos;d like to copy your citation(s) to.</p>
          <div>
            <select
              onChange={this.handleChange}
              style={{ width: '200px', marginBottom: '20px' }}
            >
              <option disabled={this.state.selectedProjectID}>-----</option>
              {this.props.projects.map(item => (
                <option
                  key={item.id}
                  id={item.id}
                  value={item.name}
                  /* Disable the current project in the dropdown */
                  disabled={item.id === this.props.currentProjectID}
                >
                  {item.name}
                </option>
              ))}
            </select>
          </div>
          <Button
            theme="primary"
            style={{ marginRight: '5px' }}
            size="lg"
            disabled={!this.state.selectedProjectID}
            onClick={this.handleSubmit}
          >
            Copy to Project
          </Button>
          <Button theme="secondary" size="lg" onClick={this.props.hideModal}>
            Cancel
          </Button>
        </center>
      </div>
    )
  }
}

const mapStateToProps = state => ({
  style: getCitationStyle(state),
  projects: state.projects,
  currentProjectID: state.currentProject.id,
})

const mapDispatchToProps = dispatch => ({
  hideModal: () => {
    dispatch(hideModal())
    dispatch(unloadCitation())
  },
  saveCitations: selectedProjectID => {
    dispatch(saveCitationsToProject(selectedProjectID))
  },
})
 
SelectProjectModal.propTypes = {
  projects: PropTypes.arrayOf(
    PropTypes.shape({
      id: PropTypes.string,
      name: PropTypes.string,
    })
  ),
  hideModal: PropTypes.func,
  saveCitations: PropTypes.func,
  currentProjectID: PropTypes.string,
}
 
export default connect(mapStateToProps, mapDispatchToProps)(SelectProjectModal)
