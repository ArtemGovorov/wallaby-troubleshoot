import React, { Component } from 'react'
import PropTypes from 'prop-types'
import { IconDone, IconError } from '../atoms/Icons'
import SimpleTable from '../atoms/SimpleTable'
 
export default class Eval extends Component {
  constructor(props) {
    super(props)
    this.state = {
      fetched: this.props.fetched || true,
      fields: [],
      found: [],
      missing: [],
    }
  }

  componentWillMount() {
    this.evaluate()
  }

  /**
   * Check for found/missing data and update state accordingly
   */
  evaluate() {
    const found = []
    const missing = []
    this.state.fields.forEach(field => {
      if (field && field.value && field.value !== '') {
        found.push(field)
      } else {
        missing.push({ label: field.label, value: 'Edit on next step' })
      }
    })
    this.setState({ found, missing })
  }

  render() {
    const cssClass = this.state.missing.length > 0 ? 'eval-2columns' : 'eval-1column'
    const foundHeader = "Here's what we found:"
    const missingHeader = "Here's what we'll need your help with:"
    return (
      this.state.fetched && (
        <span>
          <div className={cssClass}>
            <div className="summary">
              {this.state.found.length > 0 && (
                <SimpleTable
                  css="found datalist"
                  header={foundHeader}
                  items={this.state.found}
                  icon={IconDone}
                  iconCss="icon-done"
                />
              )}
            </div>
          </div>
          <div className={cssClass}>
            <div className="summary">
              {this.state.missing.length > 0 && (
                <SimpleTable
                  css="missing datalist"
                  header={missingHeader}
                  items={this.state.missing}
                  icon={IconError}
                  iconCss="icon-error"
                />
              )}
            </div>
          </div>
        </span>
      )
    )
  }
}
 
Eval.propTypes = {
  fetched: PropTypes.bool,
}
