import React, { Component } from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
import { APA_URL, MLA_URL } from 'easybib/constants'
 
import { getCitationStyle } from 'easybib/reducers'
 
import clipboard from '../utils/copypaste'
 
import Button from '../atoms/Button'
import Input from '../atoms/Input'
import Link from '../components/Link'
import { IconKeyboardArrowRight } from '../atoms/Icons'
import {
  citationTitle,
  citationYear,
  citationIdentifier,
  contributorsText,
} from '../utils/utils'

/* eslint-disable react/no-danger */
class InTextModal extends Component {
  constructor(props) {
    super(props)
    const { title, contributors, identifiers, year } = this.props
    this.state = {
      title,
      titleActive: false,
      contributors,
      contributorsActive: contributors,
      identifiers: '',
      identifiersActive: identifiers,
      year,
      yearActive: year,
      copied: false,
    }
    this.handleInputChange = this.handleInputChange.bind(this)
    this.handleCopy = this.handleCopy.bind(this)
  }

  handleCopy(html) {
    clipboard.copy({
      'text/plain': html.replace(/(<([^>]+)>)/gi, ''),
      'text/html': html,
    })
    this.setState({ copied: true })
  }

  handleInputChange(e) {
    const value = e.target.type === 'checkbox' ? e.target.checked : e.target.value
    this.setState({ [e.target.name]: value })
  }

  render() {
    const mainText = () => {
      let main = '('
      if (this.state.contributorsActive) {
        main += `${this.state.contributors}`
        if (
          this.state.titleActive ||
          this.state.identifiersActive ||
          this.state.yearActive
        ) {
          main += !this.props.style.startsWith('mla') ? ', ' : ''
        }
      }
      if (this.state.titleActive) {
        if (this.state.contributorsActive) main += ' '
        main += `<i>${this.state.title}</i>`
      }
      if (this.state.year) main += ` ${this.state.year}`
      if (this.state.identifiersActive) main += ` ${this.state.identifiers}`
      main += ')'
      return main
    }
 
    return (
      <div>
        <div className="row padded flex-center">
          <div className="col-sm-8" dangerouslySetInnerHTML={{ __html: mainText() }} />
          <div className="col-sm-4">
            <Button theme="primary" size="lg" onClick={() => this.handleCopy(mainText())}>
              {this.state.copied ? 'Copied' : 'Copy'}
            </Button>
          </div>
        </div>
        <div className="row padded-top">
          <div className="col-sm-1">
            <input
              type="checkbox"
              name="contributorsActive"
              onChange={this.handleInputChange}
              checked={this.state.contributorsActive}
            />
          </div>
          <div className="col-sm-11">
            <Input
              type="text"
              name="contributors"
              label="Contributors"
              value={this.state.contributors}
              onChange={this.handleInputChange}
            />
          </div>
        </div>
        <div className="row">
          <div className="col-sm-1">
            <input
              type="checkbox"
              name="titleActive"
              onChange={this.handleInputChange}
              checked={this.state.titleActive}
            />
          </div>
          <div className="col-sm-11">
            <Input
              type="text"
              name="title"
              label="Title"
              value={this.state.title}
              onChange={this.handleInputChange}
            />
          </div>
        </div>
        {!this.props.style.startsWith('mla') && (
          <div className="row">
            <div className="col-sm-1">
              <input
                type="checkbox"
                name="yearActive"
                onChange={this.handleInputChange}
                checked={this.state.yearActive}
              />
            </div>
            <div className="col-sm-11">
              <Input
                type="text"
                name="year"
                label="Year"
                value={this.state.year}
                onChange={this.handleInputChange}
              />
            </div>
          </div>
        )}
        <div className="row">
          <div className="col-sm-1">
            <input
              type="checkbox"
              name="identifiersActive"
              onChange={this.handleInputChange}
              checked={this.state.identifiersActive}
            />
          </div>
          <div className="col-sm-11">
            <Input
              type="text"
              name="identifiers"
              label="Identifiers"
              value={this.state.identifiers}
              onChange={this.handleInputChange}
            />
          </div>
        </div>
        {this.props.styleURL && (
          <Link orange href={this.props.styleURL} target="_blank">
            {this.props.style.toUpperCase()} parenthetical rules
            <IconKeyboardArrowRight />
          </Link>
        )}
      </div>
    )
  }
}
/* eslint-enable */
 
const mapStateToProps = state => {
  /*
    Default selected:
 
    MLA: Contributor (Last Name) separated by "and", with Identifier (Page Number)
    All other styles: Contributor (Last Name) separated by "&", with Year
 
  */
 
  const urls = {
    mla8: MLA_URL,
    mla7: MLA_URL,
    apa: APA_URL,
  }
 
  const style = getCitationStyle(state)
  return {
    style,
    citation: state.citation,
    styleURL: urls[style] || '',
    year: !style.startsWith('mla') ? citationYear(state.citation) : '',
    identifiers: style.startsWith('mla') ? citationIdentifier(state.citation) : '',
    title: citationTitle(state.citation),
    contributors: contributorsText(state.citation.contributors, style).trim(),
  }
}
 
InTextModal.propTypes = {
  title: PropTypes.string,
  contributors: PropTypes.string,
  year: PropTypes.string,
  identifiers: PropTypes.string,
  style: PropTypes.string,
  styleURL: PropTypes.string,
}
 
export default connect(mapStateToProps)(InTextModal)
