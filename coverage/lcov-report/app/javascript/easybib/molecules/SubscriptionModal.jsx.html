import React from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
 
import { Easybib } from 'subscription-frontend'

import isWindowDefined from 'easybib/utils/isWindowDefined'
import { hideModal } from 'easybib/actions/actions.modals'
import { clearEbUserAndRedirect } from 'easybib/actions/actions.redirects'
import subFeProps from 'easybib/pages/MyAccount/sharedSubscriptionFrontendProps'
 
function SubscriptionModal({ onConfirmClose, plusTest }) {
  if (!isWindowDefined) {
    return ''
  }
  // `onDismiss` + `onSuccess` are meaningful distinctions in WPT / WritingFrontend
  // In this context, they do the same thing.
  // When `hostAppLogUserOut` is deprecated (entitlement backend is better integrated)
  //  some of the callbacks may be further refactored, and make a little more
  //  sense in this context.
  return (
    <Easybib
      {...subFeProps}
      onDismiss={onConfirmClose}
      onSuccess={onConfirmClose}
      hostCodeBase="CFE"
      hostAppLogUserOut={() => clearEbUserAndRedirect('/logout')}
      plusTest={plusTest}
    />
  )
}

SubscriptionModal.propTypes = {
  onConfirmClose: PropTypes.func.isRequired,
  plusTest: PropTypes.bool,
}
 
const mapDispatchToProps = dispatch => ({
  onConfirmClose: () => dispatch(hideModal()),
})

export default connect(
  null,
  mapDispatchToProps
)(SubscriptionModal)
