class FeatureSelect extends Component {
  state = {
    activeIndex: 0,
  }
 
  setActiveIndex = index => {
    this.setState({
      activeIndex: index,
    })
  }
 
  render() {
    return (
      <React.Fragment>
        <Graphic>
          <List>
            {featureList.map((feature, index) => (
              <ListItem key={feature.title}>
                <NonButton
                  onClick={() => this.setActiveIndex(index)}
                  active={index === this.state.activeIndex}
                >
                  {feature.title}
                </NonButton>
              </ListItem>
            ))}
          </List>
          <Right>
            <Bubble>
              <ActualBubble
                style={featureList[this.state.activeIndex].position}
                dangerouslySetInnerHTML={{
                  __html: featureList[this.state.activeIndex].html,
                }}
              />
            </Bubble>
            <Sheet>
              <Caret style={featureList[this.state.activeIndex].caret_position} />
              <SheetLine />
              <SheetLine active={this.state.activeIndex === 2} />
              <SheetLine
                active={this.state.activeIndex === 0 || this.state.activeIndex === 4}
              />
              <SheetLine
                active={this.state.activeIndex === 1 || this.state.activeIndex === 3}
              />
              <SheetLine />
            </Sheet>
          </Right>
        </Graphic>
      </React.Fragment>
    )
  }
}
/* eslint-enable */
 
export default FeatureSelect
