import React, { Component } from 'react'
import PropTypes from 'prop-types'
 
import { connect } from 'react-redux'
 
import { getCitationStyle, getSourceType } from 'easybib/reducers'
import { clearSearchData } from 'easybib/actions/actions.shared'
import { checkSourceType } from 'easybib/actions/actions.search'
import { redirectToForm, redirectToSearch } from 'easybib/actions/actions.redirects'
import { manualEvent, searchEvent } from 'easybib/utils/gtm'
import { formatFormUrl } from 'easybib/utils/url'
 
import Icon from 'easybib/components/Icon'
import Link from 'easybib/components/Link'
import Input from 'easybib/components/Form/Input'
import Label from 'easybib/components/Form/Label'
import Select from 'easybib/components/Form/Select'
import { SearchSourceButton, Flex, FlexColumn, Card } from './style'
 
export class SearchSource extends Component {
  static propTypes = {
    citationStyle: PropTypes.string,
    sourceType: PropTypes.string,
    sourceTypes: PropTypes.arrayOf(
      PropTypes.shape({
        id: PropTypes.oneOfType([PropTypes.number, PropTypes.string]),
        name: PropTypes.string,
        value: PropTypes.string,
      })
    ),
    placeholder: PropTypes.string.isRequired,
    clearSearchData: PropTypes.func,
    redirectToForm: PropTypes.func,
    redirectToSearch: PropTypes.func,
    onSelectChange: PropTypes.func,
    children: PropTypes.node,
  }
 
  state = {
    text: '',
    submitted: false,
  }
 
  componentDidMount() {
    this.props.clearSearchData()
  }
 
  handleManualCitation = e => {
    e.preventDefault()
    this.props.redirectToForm(this.props.citationStyle, this.props.sourceType)
  }
 
  updateInput = e => {
    this.setState({ text: e.target.value })
  }
 
  handleSubmit = e => {
    e.preventDefault()
    if (this.state.text.trim().length !== 0) {
      this.setState({ submitted: true }, () => {
        this.props.redirectToSearch(
          this.props.citationStyle,
          this.props.sourceType,
          this.state.text
        )
      })
    }
  }
 
  render() {
    return (
      <Card id="js-cta-2">
        {this.props.children}
        <form onSubmit={this.handleSubmit} method="POST">
          <Flex>
            <FlexColumn grow="0" marginBottom="12">
              <Label htmlFor="Source type">Source type</Label>
              <Select
                options={this.props.sourceTypes}
                value={this.props.sourceType}
                onChange={this.props.onSelectChange}
                data-test="source-type"
                id="source-type"
                label="Source type"
              >
                {this.props.sourceTypes &&
                  this.props.sourceTypes.map(option => (
                    <option key={`${option.name}${option.id}`} value={option.value}>
                      {option.name}
                    </option>
                  ))}
              </Select>
            </FlexColumn>
            <FlexColumn marginBottom="12">
              <Label htmlFor="Source">Source</Label>
              <Input
                id="Source"
                placeholder={this.props.placeholder}
                value={this.state.text}
                onChange={this.updateInput}
              />
            </FlexColumn>
            <FlexColumn grow="0" marginBottom="12">
              <SearchSourceButton
                newPrimary
                fullWidth
                disabled={this.state.text.trim().length === 0 || this.state.submitted}
              >
                Cite
              </SearchSourceButton>
            </FlexColumn>
          </Flex>
        </form>
        <div>
          <Link
            gray
            onClick={this.handleManualCitation}
            href={formatFormUrl({
              citationStyle: this.props.citationStyle || 'mla8',
              sourceType: this.props.sourceType || 'website',
            })}
          >
            Manual cite
            <Icon name="rightArrow" viewBox="0 0 24 24" />
          </Link>
        </div>
        {this.props.sourceType === 'book' && (
          <img
            alt="Powered by Worldcat"
            src="http://cdn-edge2.easybib.com/images/powered_by_worldcat.png"
          />
        )}
      </Card>
    )
  }
}
 
const placeholders = {
  website: 'Cite a website by entering its URL or by searching for it.',
  book: 'Cite a book by searching by title, ISBN, or keywords.',
  journal: 'Search by article title, keywords, or DOI number.',
  film: 'Enter online video URL or film title to cite.',
  database: 'What kind of database article are you citing?',
}
 
const mapStateToProps = state => {
  const sourceTypes = state.types
    // Music and Film are considered to be duplicates of
    // "Film/Online Video" and "Musical Recording" respectively
    // so we filter them out
    .filter(el => el.name !== 'Music' && el.name !== 'Film')
    // We filter out pro sourceTypes if the user isn't a paid member
    .filter(type => !(!state.user.paidMember && type.pro))
 
  return {
    sourceTypes,
    placeholder: placeholders[getSourceType(state)] || '',
    citationStyle: getCitationStyle(state),
    sourceType: getSourceType(state),
  }
}
 
const mapDispatchToProps = dispatch => ({
  onSelectChange: e => {
    dispatch(checkSourceType(e.target.value))
  },
  redirectToForm: async (style, type) => {
    await manualEvent(style, type)
    dispatch(redirectToForm())
  },
  redirectToSearch: async (style, type, text) => {
    await searchEvent(style, type)
    dispatch(redirectToSearch(text))
  },
  clearSearchData: () => {
    sessionStorage.removeItem('ebSearchData')
    dispatch(clearSearchData())
  },
})
 
export default connect(
  mapStateToProps,
  mapDispatchToProps
)(SearchSource)
