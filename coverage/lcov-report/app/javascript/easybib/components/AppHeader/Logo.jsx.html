import React from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
import styled from 'styled-components'
import { Link } from 'react-router-dom'
 
import { getCitationStyle, getSourceType } from 'easybib/reducers'
 
export const Logo = styled.div`
  display: inline-block;
  width: 112px;
  height: 46px;
  position: relative;
  left: 52px;
  top: 8px;
  background: url(https://www.easybib.com/v2/images/easybib-chegg-logo.svg) top center
    no-repeat;
  background-size: contain;
`
 
export const LogoContainer = props =>
  props.withOutRouter ? (
    <a href="home" alt="EasyBib – a Chegg service">
      <Logo />{' '}
    </a>
  ) : (
    <Link to={props.path} alt="EasyBib – a Chegg service">
      <Logo />
    </Link>
  )
 
const mapStateToProps = state => {
  const citationStyle = getCitationStyle(state)
  const sourceType = getSourceType(state)
  return {
    path:
      citationStyle && sourceType
        ? `/style/${citationStyle}/${sourceType}-citation`
        : `/home`,
  }
}
 
LogoContainer.propTypes = {
  path: PropTypes.string,
  withOutRouter: PropTypes.bool,
}
 
export default connect(mapStateToProps)(LogoContainer)
