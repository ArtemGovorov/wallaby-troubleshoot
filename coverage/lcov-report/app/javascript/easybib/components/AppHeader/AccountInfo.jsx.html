  height: 100%;

export const AccountInfo = props => {
  const myCitationList = {
    name: 'My Citation List',
    icon: 'projects',
    href: `/project?id=${props.projectID}`,
    id: 1,
  }
 
  return (
    <AccountContainer>
      {props.hasPro && <SquareBadge>PLUS</SquareBadge>}
      <UserMenu>
        <HeaderIconAccountCircle />
        <Email>{props.email}</Email>
        <IconKeyboardArrowDown className="down-arrow-icon pull-right" />
      </UserMenu>
      <MenuRight>
        {props.projectID && <MenuItem item={myCitationList} />}
        {userItems
          .filter(item => !item.plus || (item.plus && !props.hasPro))
          .map(item => (
            <MenuItem item={item} key={item.id} />
          ))}
      </MenuRight>
    </AccountContainer>
  )
}

AccountInfo.propTypes = {
  hasPro: PropTypes.bool,
  email: PropTypes.string,
  projectID: PropTypes.string,
}
 
const mapStateToProps = state => ({
  hasPro: state.user.role === 'mybibpro',
  email: state.user.email,
  projectID: state.currentProject.id,
})
 
export default connect(mapStateToProps)(AccountInfo)
