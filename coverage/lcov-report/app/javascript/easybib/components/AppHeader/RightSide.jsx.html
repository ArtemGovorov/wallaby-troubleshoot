import React from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
import styled from 'styled-components'
import { Link } from 'react-router-dom'

import { PillBadge } from 'easybib/components/Badge'

import { tagCtaNavLink } from 'easybib/utils/gtm'

import AccountInfo from './AccountInfo'
import SignupAndLogin from './SignupAndLogin'
 
export const LinkContainer = styled.div`
  display: inline-block;
  font-size: 12px;
  font-weight: 600;
  color: #333;
  padding-right: 24px;
`
 
const NewPillBadge = styled(PillBadge)`
  @media only screen and (max-width: 755px) {
    display: none;
  }
`
 
const GrammarPlagiarismLink = styled(LinkContainer)`
  @media only screen and (max-width: 755px) {
    display: none;
  }
`

const CitationListLink = styled(LinkContainer)`
  @media only screen and (max-width: 900px) {
    display: none;
  }
`
 
const Right = styled.span`
  position: absolute;
  height: 52px;
  right: 8px;
  display: flex;
  align-items: center;
  height: 100%;
`
 
export const RightSide = props => (
  <Right>
    <React.Fragment>
      <NewPillBadge color="green">New</NewPillBadge>
      <GrammarPlagiarismLink>
        <a href="/grammar-and-plagiarism/" onClick={tagCtaNavLink}>
          {' '}
          Plagiarism and Grammar Checker
        </a>
      </GrammarPlagiarismLink>
    </React.Fragment>
    {props.projectID && (
      <CitationListLink>
        {props.withOutRouter ? (
          <a href={`/project?id=${props.projectID}`}>My Citation List</a>
        ) : (
          <Link to={`/project?id=${props.projectID}`}>My Citation List</Link>
        )}
      </CitationListLink>
    )}
    {props.loggedIn ? <AccountInfo /> : <SignupAndLogin />}
  </Right>
)
 
const mapStateToProps = state => ({
  projectID: state.currentProject.id,
  loggedIn: state.user.loggedIn,
})
 
RightSide.propTypes = {
  projectID: PropTypes.string,
  loggedIn: PropTypes.bool,
  withOutRouter: PropTypes.bool,
}
 
export default connect(mapStateToProps)(RightSide)
