interface SpinnerProps {
  page: string
  adLoader?: boolean
  showLoader?: boolean
  masterHideLoader?: () => void
  children?: React.SFC
}
 
export class AdSpinner extends Component<SpinnerProps> {
  state = { masterHide: false }
 
  timeoutID = undefined
 
  componentDidMount() {
    this.setSpinnerTimeout()
  }
 
  setSpinnerTimeout = () => {
    this.timeoutID = setTimeout(this.setMasterHide, 10000)
  }
 
  setMasterHide = () => {
    this.setState({ masterHide: true })
  }
 
  componentDidUpdate() {
    if (window.gon.cell.runLoader) {
      if (this.props.showLoader) {
        document.body.className = 'affix full-width'
      } else {
        document.body.className = ''
      }
    }
  }
 
  componentWillUnmount() {
    document.body.className = ''
    clearTimeout(this.timeoutID)
  }
 
  render() {
    const pageOptions = pages[this.props.page]
 
    const showLoader = !this.state.masterHide && this.props.showLoader
 
    return (
      <React.Fragment>
        {showLoader && (
          <Overlay>
            <Container image={pageOptions.image}>
              <Nested topOffset={pageOptions.topOffset}>
                <Balls src="/v2/images/loading-dots.gif" />
                <Text>{pageOptions.text}</Text>
              </Nested>
            </Container>
          </Overlay>
        )}
        {/* We need to use inline style's here to prevent unnecessary re-rendering */}
        <div style={{ visibility: showLoader ? 'hidden' : 'visible' }}>
          {this.props.children}
        </div>
      </React.Fragment>
    )
  }
}
 
const mapStateToProps = (state, props) => {
  const pageCustomHide = {
    results: !state.app.searchLoaded,
    form: !state.form.html,
  }
 
  return {
    showLoader:
      isWindowDefined &&
      window.gon.cell.runLoader &&
      // If we find specific page logic, we want to make sure
      // we hinge on that logic. If we don't find specific page
      // logic, we return true so that we only hinge on state.adLoader
      (state.adLoader.loading ||
        (state.adLoader.hasSeenLoader && pageCustomHide[props.page])),
  }
}
 
export default connect(mapStateToProps)(AdSpinner)
