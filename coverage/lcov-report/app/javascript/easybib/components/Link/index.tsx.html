import styled, { css } from 'styled-components'
 
interface LinkProps {
  fontSize?: string
  orange?: boolean
  blue?: boolean
  gray?: boolean
  pointer?: boolean
}
 
const Link =
  styled.a <
  LinkProps >
  `
  font-size: ${props =>
    props.fontSize ? props.fontSize : props.theme.font.text.size.sm};
  text-decoration: none;
  ${props =>
    props.orange &&
    css`
      color: ${props.theme.colors.orange};
      &:hover {
        color: ${props.theme.colors.orange};
      }
      &:focus {
        color: ${props.theme.colors.orange};
      }
    `}
  ${props =>
    props.blue &&
    css`
      color: ${props.theme.colors.blue};
      &:hover {
        color: ${props.theme.colors.blue};
      }
      &:focus {
        color: ${props.theme.colors.blue};
      }
    `}
  ${props =>
    props.gray &&
    css`
      color: ${props.theme.colors.greylighter};
      &:hover {
        color: ${props.theme.colors.orange};
      }
      &:focus {
        color: ${props.theme.colors.gray};
      }
    `}
  ${props =>
    props.pointer &&
    css`
      cursor: pointer;
    `}
  /* We'll have the icon control this eventually */
  & > svg {
    font-size: 18px;
    height: 18px;
    width: 18px;
    margin-top: -1px;
    vertical-align: middle;
  }
`
 
export default Link
