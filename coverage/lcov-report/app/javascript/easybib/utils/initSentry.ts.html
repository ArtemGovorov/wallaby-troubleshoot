import environment from 'easybib/utils/environment'
import Config from 'easybib/config'

const initRaven = () => {
  if (Config.ravenURI && window.Raven) {
    window.Raven.config(Config.ravenURI, {
      release: process.env.TRAVIS_TAG || 'travis tag not set',
      environment: environment.toLowerCase(),
      // send 50% of events, drop the other half
      sampleRate: 0.5,
      // Will only log errors from JS resources from the following hosts
      whitelistUrls: [
        // For playground & production
        /easybib\.com/,
        /d1brroc4h76ynv\.cloudfront\.net/,
        // Sentry specific
        /cdn\.getsentry\.com/,
        // Local development
        /localhost/,
      ],
    }).install()
  }
}
 
export default initRaven
