import React, { Component } from 'react'
import axios from 'axios'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
import uuidV1 from 'uuid'
import Icon from 'easybib/components/Icon'
import {
  sendMoreEvent,
  sendIntextEvent,
  tagCtaGrammar,
  tagCtaPlagiarism,
  sendShowCommentsEvent,
} from 'easybib/utils/gtm'
import { tokenStorage } from 'easybib/configureApi'
import { loadCitations } from 'easybib/actions/actions.bibliography'
import CheckboxContainer from 'easybib/components/CheckboxContainer'
import citationTypes from 'easybib/static/citationTypes'
import DOMPurify from 'dompurify'
import {
  CommentsContainer,
  CommentName,
  CommentDate,
  CommentText,
  CitationContainer,
  CitationText,
  CitationActions,
  InCitationLink,
  MaterialIcon,
  ActionButton,
  CTAContainer,
} from 'easybib/organisms/Citation/styles'
import clipboard from 'easybib/utils/copypaste'
import Button from 'easybib/atoms/Button'
import ToggleMenu from 'easybib/molecules/ToggleMenu'
import Input from 'easybib/atoms/Input'

/* eslint-disable react/no-danger */
export class Citation extends Component {
  constructor(props) {
    super(props)
    this.uuid = uuidV1()
    this.uuid_more0 = uuidV1()
    this.uuid_more1 = uuidV1()
    this.state = { showComments: false, newComment: '' }
  }

  getComments = async () => {
    this.comments = await axios.get(`${this.props.citation.id}/comments/`, {
      credentials: 'include',
      headers: {
        Accept: 'application/vnd.com.easybib.data+json',
        Authorization: `Bearer ${tokenStorage.getToken()}`,
      },
    })
    this.setState({ showComments: true })
  }
 
  handleEdit = () => {
    window.dataLayer.push({
      event: 'event',
      eventCategory: 'bibliography module',
      eventAction: 'edit',
    })
    const { projectId } = this.props
    const citationId = this.props.citation.id.substr(
      this.props.citation.id.lastIndexOf('/') + 1
    )
    const sourceType = citationTypes.find(
      el => el.apiAlias === this.props.citation.source
    )
    this.props.handleEditRedirect(projectId, citationId, sourceType.value)
  }
 
  handleDelete = () => {
    window.dataLayer.push({
      event: 'event',
      eventCategory: 'bibliography module',
      eventAction: 'delete',
    })
    this.props.handleDelete(this.props.citation)
  }

  handleCopy = () => {
    window.dataLayer.push({
      event: 'event',
      eventCategory: 'bibliography module',
      eventAction: 'copy',
    })
    clipboard.copy(this.strippedCitation())
  }

  strippedCitation = () => {
    const html = this.props.citation.formatted_data.output || ''
    // for hanging indent, it needs a p with inline style.
    const phtml = `<p style="margin-left: 1cm; text-indent: -1cm;">${html}</p>`
    return {
      'text/plain': html.replace(/(<([^>]+)>)/gi, ''),
      'text/html': phtml,
    }
  }
 
  addComment = () => {
    axios.post(
      `${this.props.citation.id}/comments/`,
      { data: { text: this.state.newComment } },
      {
        credentials: 'include',
        headers: {
          Accept: 'application/vnd.com.easybib.data+json',
          Authorization: `Bearer ${tokenStorage.getToken()}`,
        },
      }
    )
    this.setState({ showComments: false })
    this.setState({ newComment: '' })
  }
 
  handleInput = event => {
    this.setState({ newComment: event.target.value })
  }
 
  handleCommentsClick = () => {
    this.getComments()
    sendShowCommentsEvent()
  }
 
  handleCopyProjectClick = () => this.props.handleFetchProjects(this.props.citation.id)
 
  handleIntextCitationClick = () => {
    this.props.handleInText(this.props.citation, this.props.citationStyle)
    sendIntextEvent()
  }

  handleCloseClick = () => {
    this.setState({ showComments: false })
  }

  handleCheck = () => {
    if (this.props.citation.selected) {
      this.props.citation.selected = false
    } else {
      this.props.citation.selected = true
    }

    const updatedCitations = this.props.citations.map(citation => {
      if (citation.id === this.props.citation.id) {
        const newCitation = { ...citation }
        newCitation.selected = this.props.citation.selected
        return newCitation
      }
      return citation
    })
    this.props.handleSelection(updatedCitations)
  }
 
  render() {
    let classes = 'citation'
    if (this.props.first) classes += ' first'
    return (
      <CitationContainer className={classes}>
        <CheckboxContainer top={-4}>
          <input
            type="checkbox"
            id={this.props.citation.id}
            onChange={this.handleCheck}
            defaultChecked={this.props.citation.selected}
          />
          <label htmlFor={this.props.citation.id} />
        </CheckboxContainer>
        <CitationText
          dangerouslySetInnerHTML={{
            __html: DOMPurify.sanitize(this.props.citation.formatted_data.output),
          }}
        />

        <CTAContainer>
          <Button
            theme="two"
            size="lg"
            extraCss="intext"
            gtmAction="in-text citation"
            onClick={this.handleIntextCitationClick}
          >
            {this.props.citationStyle === 'chicago' ||
            this.props.citationStyle === 'turabian'
              ? 'Footnotes'
              : 'In-text Citation'}
          </Button>
          {this.props.user.loggedIn && (
            <Button
              theme="two"
              size="lg"
              extraCss="intext"
              onClick={this.handleCommentsClick}
            >
              Comments
            </Button>
          )}
          <InCitationLink href="/grammar-and-plagiarism" onClick={tagCtaGrammar}>
            Check for Grammar
          </InCitationLink>
          <InCitationLink href="/grammar-and-plagiarism" onClick={tagCtaPlagiarism}>
            Check for Plagiarism
          </InCitationLink>
        </CTAContainer>
        {this.state.showComments && (
          <div>
            <div>
              <CommentsContainer>
                <Icon className="close" name="clear" onClick={this.handleCloseClick} />
                {this.comments &&
                  this.comments.data &&
                  this.comments.data.data &&
                  this.comments.data.data.length > 0 &&
                  this.comments.data.data.map(comment => (
                    <div key={comment.data.text}>
                      <CommentName>{comment.data.fullname}</CommentName>{' '}
                      <CommentDate>{comment.data.date}</CommentDate>
                      <CommentText>{comment.data.text}</CommentText>
                    </div>
                  ))}
              </CommentsContainer>
              <Input onChange={this.handleInput} value={this.state.newComment} />
            </div>
            <Button onePress theme="primary" size="sm" onClick={this.addComment}>
              Comment
            </Button>
          </div>
        )}
 
        <CitationActions className="dropdown links">
          <ActionButton onClick={this.handleEdit}>
            <MaterialIcon name="edit" />
          </ActionButton>
          <ToggleMenu id={this.uuid} onClick={sendMoreEvent}>
            <ul className="items open-left clickable" aria-labelledby={this.uuid}>
              <li key={this.uuid_more0}>
                <Button onClick={this.handleCopy}>Copy to Clipboard</Button>
              </li>
              {this.props.user.loggedIn && (
                <li>
                  <Button onClick={this.handleCopyProjectClick}>
                    Copy to Another Project
                  </Button>
                </li>
              )}
              <li key={this.uuid_more1}>
                <Button onClick={this.handleDelete}>Delete</Button>
              </li>
            </ul>
          </ToggleMenu>
        </CitationActions>
      </CitationContainer>
    )
  }
}
/* eslint-enable */
 
Citation.propTypes = {
  projectId: PropTypes.string,
  citation: PropTypes.shape({
    id: PropTypes.string,
    formatted_data: PropTypes.shape({
      output: PropTypes.string,
    }),
    source: PropTypes.string,
    citations: PropTypes.string,
    type: PropTypes.string,
    selected: PropTypes.bool,
  }),
  citations: PropTypes.arrayOf(
    PropTypes.shape({
      text: PropTypes.string,
      type: PropTypes.string,
    })
  ),
  citationStyle: PropTypes.string,
  handleInText: PropTypes.func,
  handleDelete: PropTypes.func,
  handleEditRedirect: PropTypes.func,
  first: PropTypes.bool,
  user: PropTypes.shape({
    loggedIn: PropTypes.bool,
  }),
  handleFetchProjects: PropTypes.func,
  handleSelection: PropTypes.func,
}

const mapStateToProps = state => ({
  citations: state.citations || [],
})
 
const mapDispatchToProps = dispatch => ({
  handleSelection: citations => {
    dispatch(loadCitations(citations))
  },
})
 
export default connect(
  mapStateToProps,
  mapDispatchToProps
)(Citation)
