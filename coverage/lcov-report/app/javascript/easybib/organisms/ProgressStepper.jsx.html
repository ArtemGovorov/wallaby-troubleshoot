import React, { Component, Fragment } from 'react'
import PropTypes from 'prop-types'
import cc from 'classcat'
 
const steps = [
  { name: 'Select your Source', progress: { big: '-11.8', small: '13.2' } },
  { name: 'Review info', progress: { big: '38.2', small: '38.2' } },
  { name: 'Add missing info', progress: { big: '63.2', small: '63.2' } },
  { name: 'Get Citation' },
]
 
class ProgressStepper extends Component {
  constructor() {
    super()
    this.state = {
      headerIsActive: false,
      width: null,
    }
    this.handleScroll = this.handleScroll.bind(this)
    this.handleResize = this.handleResize.bind(this)
  }

  componentDidMount() {
    window.addEventListener('scroll', this.handleScroll)
    window.addEventListener('resize', this.handleResize)
  }
 
  componentWillUnmount() {
    window.removeEventListener('scroll', this.handleScroll)
    window.removeEventListener('resize', this.handleResize)
  }

  handleScroll() {
    const element = this.progressStepperSticky
    const elementBottom = element.getBoundingClientRect().bottom
 
    const container = document.getElementById('main-content')
 
    if (elementBottom <= 0) {
      this.setState({
        headerIsActive: true,
        width: container.getBoundingClientRect().width,
      })
    } else if (elementBottom >= 0) {
      container.removeAttribute('style')
      this.setState({
        headerIsActive: false,
      })
    }
  }
 
  handleResize() {
    this.handleScroll()
    const mainElement = document.getElementById('main-content')
    this.setState({
      width: mainElement.getBoundingClientRect().width,
    })
  }

  render() {
    return (
      <Fragment>
        <div
          ref={div => {
            this.progressStepperSticky = div
          }}
        />
        <div
          className={cc({
            'progress-stepper-outer': true,
            fixed: this.state.headerIsActive,
          })}
          style={{ width: this.state.headerIsActive ? this.state.width : 'auto' }}
        >
          <div className="progress-stepper">
            <div className="progress">
              {['big', 'small'].map(item => (
                <div
                  key={item}
                  aria-valuemax="100"
                  aria-valuemin="0"
                  aria-valuenow="13.2"
                  className={cc({
                    'progress-bar': true,
                    smaller: item === 'small',
                  })}
                  style={{ width: `${steps[this.props.step - 1].progress[item]}%` }}
                />
              ))}
            </div>
            {steps.map((item, i) => (
              <div
                key={item.name}
                className={`progress-step-item ${
                  i === this.props.step - 1 ? 'active' : ''
                } ${i < this.props.step - 1 ? 'done' : ''} ${i === 0 ? 'first' : ''} ${
                  i === steps.length - 1 ? 'last' : ''
                }`}
              >
                <div className="progress-ui">
                  <div className="round-number" />
                </div>
                <div className="progress-desc">
                  <div className="progress-desc-inner" role="button">
                    {item.name}
                    <span className="progress-stepof">
                      (Step {i + 1} of {steps.length})
                    </span>
                  </div>
                </div>
              </div>
            ))}
          </div>
        </div>
      </Fragment>
    )
  }
}

ProgressStepper.propTypes = {
  step: PropTypes.number.isRequired,
}
 
export default ProgressStepper
