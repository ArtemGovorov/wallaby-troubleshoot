import React from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
 
import { hideModalAfterTimeout } from 'easybib/actions/actions.modals'
import CreateProjectModal from '../molecules/CreateProjectModal'
import SelectProjectModal from '../molecules/SelectProjectModal'
import FootnoteModal from '../molecules/FootnoteModal'
import InTextModal from '../molecules/InTextModal'
import Modal from '../molecules/Modal'
import CancellationAlertModal from '../molecules/CancellationAlertModal'
import SubscriptionModal from '../molecules/SubscriptionModal'
 
function ModalWrapper({ type, hideAfterTimeout, options }) {
  let node
  let heading
  let noHeader = false
  let customDialogWidth
  let customBodyPadding
  let customBodyHeight
  switch (type) {
    case 'loader':
      node = <div>Loader</div>
      heading = 'Loader'
      break
    case 'intext-citation':
      node = <InTextModal />
      heading = 'In-text citation'
      break
    case 'footnote':
      node = <FootnoteModal />
      heading = 'Footnote'
      break
    case 'select-project':
      node = <SelectProjectModal />
      noHeader = true
      break
    case 'create-project':
      node = <CreateProjectModal />
      noHeader = true
      break
    case 'cancellation-alert':
      node = (
        <CancellationAlertModal text="Your cancellation is being processed and will complete shortly." />
      )
      hideAfterTimeout({
        modalType: type,
        seconds: 2.5,
      })
      noHeader = false
      break
    case 'fetch-cancellation-error':
      node = (
        <CancellationAlertModal text="It's not you, it's us! Your request has errored, please try again." />
      )
      noHeader = false
      break
    case 'subscription':
      node = <SubscriptionModal plusTest={options.plusTest} />
      noHeader = true
      customDialogWidth = '800px'
      customBodyPadding = '0px'
      customBodyHeight = '100vh'
      break
    default:
      node = <span />
  }
 
  return (
    <Modal
      heading={heading}
      noHeader={noHeader}
      customDialogWidth={customDialogWidth}
      customBodyPadding={customBodyPadding}
      customBodyHeight={customBodyHeight}
    >
      {node}
    </Modal>
  )
}

ModalWrapper.propTypes = {
  type: PropTypes.string,
  hideAfterTimeout: PropTypes.func.isRequired,
  options: PropTypes.shape({ plusTest: PropTypes.bool }),
}
 
const mapStateToProps = state => ({
  type: state.modal.modalType,
  options: state.modal.options,
})

const mapDispatchToProps = {
  hideAfterTimeout: hideModalAfterTimeout,
}
 
export default connect(mapStateToProps, mapDispatchToProps)(ModalWrapper)
