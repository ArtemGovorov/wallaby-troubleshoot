import React from 'react'
import PropTypes from 'prop-types'
import Button from '../atoms/Button'
import Contributors from '../atoms/Contributors'
import SourceDescription from '../atoms/SourceDescription'
import SourceLink from '../atoms/SourceLink'
import SourceTitle from '../atoms/SourceTitle'

function SearchResultCard({ item, handleCite, searchText, searchType, style }) {
  const handleClick = () => {
    handleCite(item, searchText, searchType, style)
  }
 
  return (
    <section className="search-result-card">
      <div className="row">
        <div className="col-xs-12">
          <div className="text">
            <h1>
              <SourceTitle source={item} />
            </h1>
            <p>
              <Contributors source={item} />
              <SourceDescription
                source={item}
                type={searchType}
                hasContributors={
                  item.data.contributors && item.data.contributors.length > 0
                }
              />
            </p>
            <SourceLink source={item} />
          </div>
          <div className="action">
            <Button
              onePress
              theme="primary"
              size="lg"
              extraCss="cite"
              onClick={handleClick}
            >
              Cite
            </Button>
          </div>
        </div>
      </div>
    </section>
  )
}
 
SearchResultCard.propTypes = {
  item: PropTypes.shape({
    data: PropTypes.object,
    display: PropTypes.object,
  }).isRequired,
  handleCite: PropTypes.func.isRequired,
  searchText: PropTypes.string,
  searchType: PropTypes.string,
  style: PropTypes.string,
}
 
export default SearchResultCard
