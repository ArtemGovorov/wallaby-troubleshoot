import React, { Component } from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
import DOMPurify from 'dompurify'
import withReduxProvider from 'easybib/hocs/withReduxProvider'
import { typesRedirect } from 'easybib/utils/utils'
import { isManualForm } from 'easybib/utils/url'
import { getSourceType } from 'easybib/reducers'
import Loading from 'easybib/components/Loading'
import SearchSourceSmall from 'easybib/pages/Form/SearchSourceSmall'
 
import Button from '../atoms/Button'
import SavingModal from '../molecules/SavingModal'
import {
  IconKeyboardArrowLeft,
  IconKeyboardArrowRight,
  IconAddCircle,
  IconDelete,
} from '../atoms/Icons'

import {
  fetchLegacyFormJS,
  initLegacyFormJS,
  formSubmit,
  unloadFormData,
} from '../actions/actions.form'
 
import { redirectToSearch } from '../actions/actions.redirects'
 
const FormButtons = ({ citationIDExists }) => {
  const buttonText = {
    back: citationIDExists ? 'Back to my Citation List' : 'Back to Review',
    next: citationIDExists ? 'Update Citation' : 'Complete Citation',
  }
  return (
    <div className="btn-wrapper">
      <Button
        theme="tertiary"
        size="lg"
        onClick={() => {
          window.history.back()
        }}
        type="button"
      >
        <IconKeyboardArrowLeft />
        {buttonText.back}
      </Button>
      <Button theme="primary" size="lg" extraCss="complete-citation" type="submit">
        {buttonText.next}
        <IconKeyboardArrowRight />
      </Button>
    </div>
  )
}
 
FormButtons.propTypes = {
  citationIDExists: PropTypes.bool,
}
 
class Form extends Component {
  state = {
    legacyFormJSInitialized: false,
  }
 
  componentDidMount() {
    this.props.fetchLegacyFormJS()
    this.initForm()
  }
 
  componentDidUpdate() {
    this.initForm()
  }
 
  componentWillUnmount() {
    this.props.unloadFormData()
  }
 
  initForm = () => {
    if (!this.state.legacyFormJSInitialized && this.props.html !== '') {
      // Once the form has initialized, it won't attempt to
      // do so again
      this.setState({ legacyFormJSInitialized: true })
      // This html has to be appended to the document manually
      // to guarantee it's presence in the DOM before the legacy form js
      // does DOM manipulations on it (and just because this.props.html
      // is true, doesn't mean it's necessarily available in the DOM yet)
      this.formHTML.innerHTML = DOMPurify.sanitize(this.props.html)
      this.props.initLegacyFormJS()
    }
  }
 
  handleSubmit = e => {
    e.preventDefault()
 
    // unlock modal uses window.easybib_form_no_submit
    // they use this to prevent submission of the form,
    // which they later do by manually triggering the submit
    // button later.
    if (window.easybib_form_no_submit !== true) {
      this.props.formSubmit()
    }
  }
 
  formHTMLRef = node => {
    this.formHTML = node
  }
 
  render() {
    const contentVisibility = this.props.html ? 'visible' : 'hidden'
    const loaderVisibility = this.props.html ? 'hidden' : 'visible'

    return (
      <span>
        <IconAddCircle className="icon-add_circle hidden" />
        <IconDelete className="icon-delete hidden" />
        <div style={{ visibility: contentVisibility }}>
          {this.props.showSearchComponent && (
            <SearchSourceSmall type={this.props.type} onSearch={this.props.redirectTo} />
          )}
          <form id="citation_form" onSubmit={this.handleSubmit} method="POST">
            <div className="form-wrapper">
              <div className="form-outer">
                <div id="form-inner">
                  <div ref={this.formHTMLRef} />
                </div>
              </div>
            </div>
            <SavingModal visible={this.props.modalVisible} />
            <FormButtons citationIDExists={this.props.citation.id} />
          </form>
        </div>
        <div style={{ visibility: loaderVisibility }}>
          <Loading message="One moment while we gather the info for your citation..." />
        </div>
      </span>
    )
  }
}
 
Form.propTypes = {
  fetchLegacyFormJS: PropTypes.func,
  initLegacyFormJS: PropTypes.func,
  unloadFormData: PropTypes.func,
  showSearchComponent: PropTypes.bool,
  citation: PropTypes.shape({
    id: PropTypes.string,
    callback: PropTypes.func,
    onSubmit: PropTypes.func,
    links: PropTypes.shape({
      me: PropTypes.string,
    }),
  }),
  html: PropTypes.string,
  formSubmit: PropTypes.func,
  type: PropTypes.string,
  redirectTo: PropTypes.func,
  modalVisible: PropTypes.bool,
}

const mapStateToProps = state => ({
  html: state.form.html,
  citation: state.citation,
  type: getSourceType(state),
  showSearchComponent: !typesRedirect[state.currentSourceType] && isManualForm(),
  modalVisible: state.modal.visible,
})
 
const mapDispatchToProps = {
  fetchLegacyFormJS,
  initLegacyFormJS,
  redirectTo: redirectToSearch,
  unloadFormData,
  formSubmit,
}
 
export default connect(
  mapStateToProps,
  mapDispatchToProps
)(withReduxProvider(Form))
