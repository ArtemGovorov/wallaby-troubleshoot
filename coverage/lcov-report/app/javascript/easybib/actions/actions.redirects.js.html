import * as constants from 'easybib/constants'
 
import { getCitationStyle, getSourceType } from 'easybib/reducers'
 
import { deleteEbUser } from 'easybib/utils/localStorage'

import {
  formatFormUrl,
  formatSearchUrl,
  formatStyleUrl,
  formatHomeUrl,
} from 'easybib/utils/url'
 
import { push } from 'easybib/utils/history'
 
export const redirect = url => {
  window.location.href = url
}
 
export const redirectToHomePage = () => (dispatch, getState) => {
  const citationStyle = getCitationStyle(getState())
  const sourceType = getSourceType(getState())
  const homeUrl =
    citationStyle && sourceType
      ? formatStyleUrl({ citationStyle, sourceType })
      : formatHomeUrl()
  push(homeUrl)
}

export const redirectToStylePage = () => () => {
  const url = formatStyleUrl({})
  push(url)
}

export const redirectToForm = newOrManual => (dispatch, getState) => {
  const citationStyle = getCitationStyle(getState())
  const sourceType = getSourceType(getState())
  const url = formatFormUrl({ citationStyle, sourceType, newOrManual })
  push(url)
}

export const redirectToEditForm = (projectID, citationID, type) => (
  dispatch,
  getState
) => {
  const citationStyle = getCitationStyle(getState())
  const sourceType = type || getSourceType(getState())
  const url = formatFormUrl({ citationStyle, sourceType, projectID, citationID })
  push(url)
}

/**
 * Check if user has pro role
 * @param {String} role
 * @return {Bool}
 */
export const userIsPro = role => {
  if (!role) return false
  const notLoggedIn = ['', 'not_logged_in'].includes(role)
  const freeAccount = ['mybib'].includes(role)
  const notProAccount = notLoggedIn || freeAccount

  return !notProAccount
}

export const nonProRedirect = () => (dispatch, getState) => {
  const isPro = userIsPro(getState().user.role)
  if (isPro) return Promise.resolve()
  redirect(constants.PRICING_PATH)
  // We can refactor this to return an error in the future
  // by using .catch() off of usage of nonProRedirect
  // eslint-disable-next-line prefer-promise-reject-errors
  return Promise.reject('Non Pro Redirect')
}
 
export const clearEbUserAndRedirect = link => {
  deleteEbUser()
  redirect(link)
}
 
export const redirectToBibliographyPage = ({ style, projectId, citationAdded }) => (
  dispatch,
  getState
) => {
  const projId = projectId || getState().currentProject.id
  let url = `/project`
  if (style) url += `/style/${style}`
  url += `?id=${projId}`
  if (citationAdded) url += '&new=1'
  push(url)
}
 
export const redirectToSearch = term => (dispatch, getState) => {
  const citationStyle = getCitationStyle(getState())
  const sourceType = getSourceType(getState())
  const searchText = term || getState().searchData.searchText
  const url = formatSearchUrl({
    citationStyle,
    sourceType,
    searchText,
  })
  push(url)
}
