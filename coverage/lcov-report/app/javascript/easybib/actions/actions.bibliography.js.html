/**
 * Update|save a bibliography default style.
 * @param {object} project - bibliography
 * @param {string} style
 * @uses dataApi.updateProjectStyle()
 */
const updateProjectStyle = (project, style) => (dispatch, getState, { dataApi }) => {
  if (style === project.defaultstyle) return Promise.resolve()
  return dataApi.updateProjectStyle(project, style)
}

/**
 * To signal to GA that there is a new citation added,
 *   redirectToBibliographyPage adds ?new=1 onto the url. However, people might
 *   then come back to this with their browser which would cause GA to think
 *   there is a new citation.
 * We remove the ?new=1 from the browser url, so user doesn't use this new=1 url.
 *
 * https://developer.mozilla.org/en-US/docs/Web/API/History_API#The_replaceState()_method
 */
export const updateBibliographyUrl = () => {
  window.history.replaceState('notnew', '', window.location.href.replace('&new=1', ''))
}
 
export const checkIfNewCitation = () => (dispatch, getState) => {
  if (window.location.href.includes('&new=1')) {
    formEventCreate(getCitationStyle(getState()), getSourceType(getState()), 'success')
    updateBibliographyUrl()
  }
}
 
/**
 * Get footnote for a given citation.
 * Documentation for footnote API
 * http://developer.easybib.com/citation-formatting-api/citation-formatting-api-api-spec-v2-1/
 * @param {object} citation
 */
export const getFootnote = citation => (dispatch, getState, { formatApi }) => {
  const data = { type: 'footnotes', ...citation }
  return formatApi
    .formatCitations([data], aliasToCode(getCitationStyle(getState())))
    .then(
      response => {
        const footnote = response.data[0].citation
        const citationId = response.data[0].custom_id
        dispatch(updateStoreCitation(citationId, { footnote }))
        // update current citation
        const updatedCitation = getState().citations.find(c => c.id === citationId)
        dispatch(loadCitation(updatedCitation))
      },
      e => {
        dispatch(hideModal())
        dispatch(displayError(constants.GET_FOOTNOTE_ERROR, e))
      }
    )
}

/**
 * Load bibliography and its citations on BibliographyPage and format them according to the
 * bibliography default style if needed.
 * @param {string} projectId - bibliography id
 * @param {string} style
 * @uses dataApi.getCitations()
 */
export const loadProjectAndCitations = () => (dispatch, getState, { dataApi }) => {
  const project = getState().currentProject
  const style = getCitationStyle(getState())
  const styleInUrl = window.location.href.includes('style')
  const citationStyle = styleInUrl ? style || project.defaultstyle : project.defaultstyle
  // If the user could not retrieve a project for whatever reason,
  // clear their token cache, and redirect to homepage
  if (!project || Object.keys(project).length === 0) {
    deleteEbUser()
    return redirectToHomePage(push)
  }
 
  dispatch(setCitationStyle(citationStyle))
  return dispatch(updateProjectStyle(project, aliasToCode(citationStyle))).then(
    () =>
      dataApi.getCitations(project, aliasToCode(citationStyle)).then(
        resp => {
          if (resp.data && resp.data.length > 0) {
            const normalizedCitations = sortCitations(normalize(resp.data))

            dispatch(loadCitations(normalizedCitations))
          } else {
            dispatch(loadCitations([]))
          }
        },
        e => dispatch(displayError(constants.GET_CITATIONS_ERROR, e))
      ),
    e => dispatch(displayError(constants.UPDATE_PROJECT_ERROR, e))
  )
}
 
/**
 * Load additional citation styles to state.styles, with a merge|overwrite option.
 * Overwrite is typically used when searching for extra styles, when we want to display only search
 * results in the styles dropdown.
 * Used in BibliographyPage.
 * Stored in state.styles
 * @param {string} term - searched term
 * @param {integer} page - page from searched results
 * @param {boolean} merge - merge or overwrite current styles with search results
 * @param {function} callback
 */
export const loadMoreStyles = (term, page, merge = false, callback) => (
  dispatch,
  getState,
  { formatApi }
) => {
  if (!merge) dispatch(searchStylesStart())
  formatApi.getStyles(term, page).then(styles => {
    if (merge) {
      dispatch(addStyles(styles.results))
    } else {
      dispatch(searchStylesEnd())
      dispatch(loadStyles(styles.results))
    }
    if (callback && typeof callback === 'function') {
      callback(styles.results)
    }
  })
}
 
const getProjects = () => (dispatch, getState, { dataApi }) =>
  new Promise(resolve => {
    // Make a call for projects, only if we don't currently have any.
    // Resolve once we receive them
    if (!getState().projects.length) {
      dataApi.getProjects().then(({ data }) => {
        dispatch(addProjects(normalize(data)))
        resolve()
      })
      // If we DO have projects, resolve right away
    } else {
      resolve()
    }
  })

export const showCitationModals = citationID => (dispatch, getState) => {
  if (citationID) {
    dispatch(loadCitation(citationID))
  }
  dispatch(getProjects()).then(() => {
    const moreThanOneProject = getState().projects.length > 1
    if (moreThanOneProject) {
      dispatch(showModal('select-project'))
    } else {
      dispatch(showModal('create-project'))
    }
  })
}

const reduxCitationAPIHackyNormalize = item => ({ data: item })

/**
 * Save citations to a project:
 * - if there is a citation in state.citation we use that one
 * - if not we use the ones from the current project in state.citations
 * @param {String} projectId
 * @return {Promise<Object>}
 */
export const saveCitationsToProject = projectID => (dispatch, getState, { dataApi }) => {
  const project = getProject(getState(), projectID)
  const citations = getCitations(getState())
 
  const selectedCitations = getSelectedCitations(getState())
  const selectedCitation = getSelectedCitation(getState())
 
  let citationsDataList
  if (selectedCitation) {
    citationsDataList = [selectedCitation]
  } else if (selectedCitations.length) {
    citationsDataList = selectedCitations
  } else citationsDataList = citations
 
  const formattedData = citationsDataList.map(item =>
    reduxCitationAPIHackyNormalize(item)
  )

  return dataApi
    .addCitationsToProject(project, formattedData)
    .then(() => redirect(`/project?id=${projectID}`))
}

/**
 * Create a new bibliography|project.
 * @param {string} style - bibliography style
 * @uses dataApi.createProject()
 * @todo take `createProject` and this function, and refactor into something
 * a little bit cleaner
 */
export const createNamedProject = name => (dispatch, getState, { dataApi }) => {
  const style = getState().currentProject.defaultstyle
  const shouldProjectBePublic = getState().user.loggedIn === false
  return dataApi.createProject(name, style, shouldProjectBePublic).then(
    response => {
      const project = mergeDataAndLinks(response)
      const currentProjects = getState().projects
      const combinedProjects = currentProjects.concat(project)
      dispatch(addProjects(combinedProjects))
      dispatch(setCurrentProject(combinedProjects[0]))
      dispatch(saveCitationsToProject(response.data.id))
    },
    e => dispatch(displayError(constants.CREATE_PROJECT_ERROR, e))
  )
}

export const initBibliographyPage = () => async dispatch => {
  await dispatch(initApp('bibliography'))
  dispatch(loadProjectAndCitations())
  dispatch(checkIfNewCitation())
}
