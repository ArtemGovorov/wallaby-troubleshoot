import { setCurrentProject, mergeDataAndLinks } from 'redux-citation-api'
import * as constants from 'easybib/constants'
import { redirectToHomePage } from './actions.redirects'
import { clearProjectID } from './actions.shared'
import { displayError } from './actions.errors'
import { setLocal, deleteLocalProjectID } from '../utils/localStorage'
import { trackWithoutThrow } from '../utils/track'
import { EasybibThunkAction } from 'types/easybibRedux'
 
/**
 * Check citation count for current user if logged in.
 *
 * @param {String} projectId - Current project id
 * @return {Promise}
 */
const setCitationCount = (): EasybibThunkAction<Promise<number | void>> => async (
  dispatch,
  getState,
  { dataApi }
) => {
  const { projectID } = getState()
  if (!projectID) {
    return Promise.resolve()
  }
  return dataApi.getProjectWithCount(projectID).then(
    project => {
      dispatch(setCurrentProject(mergeDataAndLinks(project)))
      setLocal('citationCount', project.data.citationCount)
      return project.data.citationCount
    },
    e => {
      const message = e.message && e.message.msg
      if (
        (message && message.includes('Not enough permissions for this project')) ||
        message.includes('The project you requested cannot be found')
      ) {
        trackWithoutThrow(e, getState().user, {
          message: 'actions.adds error permissions',
        })
        dispatch(clearProjectID())
        deleteLocalProjectID()
        dispatch(redirectToHomePage())
      } else {
        trackWithoutThrow(e, getState().user, {
          message: 'actions.adds error unclear',
        })
        dispatch(displayError(constants.CITATIONS_COUNT_ERROR, e))
      }
    }
  )
}

export default setCitationCount
