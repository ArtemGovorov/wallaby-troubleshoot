import { setCurrentProject, mergeDataAndLinks } from 'redux-citation-api'
import * as constants from 'easybib/constants'
import { fetchUser } from 'easybib/utils/utils'
import { displayError } from './actions.errors'
import { redirectToHomePage } from './actions.redirects'
import requestWithCredentials from '../utils/api'
import { EasybibThunkAction } from 'types/easybibRedux'
const currentDate = new Date().toLocaleDateString()
 
/**
 * Create a new bibliography|project.
 * @param {string} style - bibliography style
 * @uses dataApi.createProject()
 */
const createProject = (style: string): EasybibThunkAction<void> => (
  dispatch,
  getState,
  { dataApi }
) => {
  const shouldProjectBePublic = getState().user.loggedIn === false
  dataApi
    .createProject(
      `${constants.PROJECT_NAME} ${currentDate}`,
      style,
      shouldProjectBePublic
    )
    .then(
      async response => {
        const project = mergeDataAndLinks(response)
        await requestWithCredentials('/api/session_project', 'POST', {
          projectId: project.id,
        })
        await fetchUser()
        dispatch(setCurrentProject(project))
      },
      e => dispatch(displayError(constants.CREATE_PROJECT_ERROR, e))
    )
}
 
/**
 * Initlialize|load bibliography or create a new one if needed.
 * Used on FormPage
 * @param {string} style
 * @param  {string} projectId - bibliography id
 * @uses dataApi.getProject()
 */
export const initProject = (
  style: string = 'mla8',
  projectId: string
): EasybibThunkAction<void> => (dispatch, getState, { dataApi }) => {
  if (!projectId) return dispatch(createProject(style))

  return dataApi.getProject(projectId).then(
    projectData => {
      const project = mergeDataAndLinks(projectData)
      dispatch(setCurrentProject(project))
    },
    e => {
      if (
        e.message &&
        e.message.msg &&
        e.message.msg.includes('Not enough permissions for this project')
      ) {
        redirectToHomePage()
      } else dispatch(displayError(constants.GET_PROJECT_ERROR, e))
    }
  )
}
 
export default initProject
