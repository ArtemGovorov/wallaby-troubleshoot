import * as constants from 'easybib/constants'
import { initApp } from './actions.shared'
import { loadSearchData, storeSearchData, loadCitation } from './actions.search'
import { displayError, pageNotFound } from './actions.errors'
import { cslToEB, autociteUrl, sendMetrics } from '../utils/AutociteUtils'

/**
 * Evaluate found/missing data before citation creation (Eval page).
 * If searched source is website, grab citation from autocite.
 */
export const evalSearchData = () => (dispatch, getState) => {
  dispatch(loadSearchData())
  const { searchData, user } = getState()
  if (searchData.searchType === 'website') {
    const url =
      searchData.source.data.autocite.search || searchData.source.data.autocite.url
    return autociteUrl(url).then(
      cslResponse => {
        sendMetrics(cslResponse, 'eval', user)
        const response = cslToEB(cslResponse)
        searchData.source.data = response.length ? response[0].data : null
        dispatch(storeSearchData(searchData))
        dispatch(loadCitation(searchData.source.data))
      },
      e => {
        dispatch(displayError(constants.GET_CITATION_ERROR, e))
      }
    )
  }
  return searchData.source
    ? dispatch(loadCitation(searchData.source.data))
    : dispatch(pageNotFound())
}

export const initEvalPage = () => async dispatch => {
  await dispatch(initApp('eval'))
  dispatch(evalSearchData())
}
