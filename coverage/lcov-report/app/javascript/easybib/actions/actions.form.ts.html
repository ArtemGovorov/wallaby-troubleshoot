/**
 * @param {String|Object} e
 * @return {String}
 */
export const errorFromRequest = e => {
  let error = constants.ADD_CITATION_ERROR
 
  const msg =
    typeof e.message === 'object' ? e.message.message || e.message.msg : e.message || ''

  if (msg === constants.URL_EMPTY_ERROR) {
    error = constants.REQUIRED_URL_ERROR
  } else if (msg.includes(constants.CURL_ERROR)) {
    error = constants.URL_NORESULTS_ERROR
  }
  return error
}

/**
 * Get a citation.
 * @param {string} projectId - bibliography id
 * @param {string} citationId
 * @return {Promise}
 * @uses dataApi.getCitation()
 */
export const getCitation = (projectId, citationId): EasybibThunkAction<Promise<void>> => (
  dispatch,
  getState,
  { dataApi }
) => {
  if (!citationId) return Promise.resolve()

  return dataApi.getCitation(projectId, citationId).then(
    response => {
      const citation = mergeDataAndLinks(response)
      dispatch(loadCitation(citation))
    },
    e => dispatch(displayError(constants.GET_CITATION_ERROR, e))
  )
}

export const cleanupCitation = citation => {
  const cleanCitation = { ...citation }
 
  if (citation.pubtype && citation.pubtype.suffix === '') {
    delete cleanCitation.pubtype.suffix
  }
  if (citation.pubtype && citation.pubtype.main === '') {
    delete cleanCitation.pubtype.main
  }
  if (!citation[citation.source]) {
    cleanCitation[citation.source] = {}
  }
 
  // remove all the other pubtypes and sources
  const whitelisted = key => {
    const allowed = [
      'source',
      'pubtype',
      'contributors',
      'doi',
      'annotation',
      'email',
      'issn',
      'keywords',
      'oclc',
      'npnd',
      citation.source,
    ]
    if (citation.pubtype && citation.pubtype.main) {
      allowed.push(citation.pubtype.main)
    }
    if (citation.pubtype && citation.pubtype.suffix) {
      allowed.push(citation.pubtype.suffix)
    }
    return allowed.indexOf(key) !== -1
  }
 
  Object.keys(citation).forEach(key => {
    if (!whitelisted(key)) delete cleanCitation[key]
  })
  return cleanCitation
}
 
/**
 * Update a citation (FormPage manual).
 * @param {string} citationLink
 * @param {object} citationData
 * @uses dataApi.saveCitation()
 */
export const updateCitation = (
  citationLink: string,
  citationData: { source: string }
): EasybibThunkAction<Promise<void>> => (dispatch, getState, { dataApi }) => {
  const uuid = dispatch(createLoading())
  const citation = cleanupCitation(citationData)
  window.gon.disableCreateCitationForm()
  window.scrollTo(0, 0)
  dispatch(showModal())
  return dataApi.saveCitation(citationLink, citation).then(
    () => {
      if (!window.gon.cell || !window.gon.cell.runLoader) dispatch(loadingHide(uuid))
      const project = getState().currentProject
      const style = project.defaultstyle
      dispatch(hideModal())
      formEventUpdate(style, citationData.source, 'success', () =>
        dispatch(
          redirectToBibliographyPage({
            style,
            projectId: project.id,
            citationAdded: false,
          })
        )
      )
    },
    e => {
      window.gon.enableCitationForm()
      formEventUpdate(
        getState().currentProject.defaultstyle,
        citationData.source,
        'error'
      )
      dispatch(displayError(constants.SAVE_CITATION_ERROR, e))
      dispatch(hideModal())
      trackWithoutThrow(e, getState().user, {
        state: getState(),
        error: e,
        message: 'update citation error',
      })
    }
  )
}

/**
 * Add a citation to a bibliography.
 * @param {object} citationData
 * @param {bool} manual - Manual citation creation
 * @uses redux-citation-api.addCitation()
 */
export const createCitation = ({ citationData, manual = false }) => (
  dispatch,
  getState
) => {
  const uuid = dispatch(createLoading())
  const citation = { data: { ...citationData } }
  window.gon.disableCreateCitationForm()
  const autociteData = getState().searchData
  if (autociteData && autociteData.source && autociteData.searchType === 'website') {
    sendAutociteFeedback(citationData, autociteData.source, getState().user)
  }
  window.scrollTo(0, 0)
  const notPro = getState().user.role !== 'mybibpro'
  dispatch(showModal())
  dispatch(addCitation(citation, true, manual)).then(
    () => {
      const project = getState().currentProject
      const style = urlParams('form').style || getState().style
      const addNewCitation = true
      formEventViaDispatch(style, citationData.source)
      if (!window.gon.cell || !window.gon.cell.runLoader) dispatch(loadingHide(uuid))
      dispatch(hideModal())
      dispatch(
        redirectToBibliographyPage({
          style,
          projectId: project.id,
          citationAdded: addNewCitation,
        })
      )
    },
    e => {
      window.gon.enableCitationForm()
      const error = errorFromRequest(e)
      dispatch(displayError(error, e))
      dispatch(hideModal())
      trackWithoutThrow(e, getState().user, {
        state: getState(),
        error: e,
        message: 'create citation error',
      })
    }
  )
}
 
export const initFormForPage = (type, style) => (dispatch, getState) => {
  const isEditForm = () => {
    // getCitation puts the citation in the state when it's an edit form.
    const { citation } = getState()
    return citation !== undefined && Object.keys(citation).length !== 0
  }
  // no data should be put in the form if it is a manualForm.
  if (!window.gon.manualForm && !isEditForm()) {
    dispatch(loadSearchData())
    const { searchData } = getState()
    if (window.citationFormData) {
      dispatch(loadCitation(window.citationFormData))
    } else if (searchData.source) {
      dispatch(loadCitation(searchData.source.data))
    }
  }

  const verifiedStyle = dispatch(verifyUrlStyle(style))

  requestForm(type, verifiedStyle).then(response => {
    dispatch(loadFormHtml(response.form))
  })
  dispatch(setCitationStyle(verifiedStyle))
}
 
export const initFormPage = () => async (dispatch, getState) => {
  const formAPIUrl = getFormAPIUrl()
  const formPageParams = urlParams('form')
  const backendUser = await requestWithCredentials(formAPIUrl)
  await dispatch(initApp('form'))
 
  /**
   * Be sure to make the API requests first if needed
   * resp.projectId and citationId come from URL parameters when editing an existing citation:
   * /$STYLE/$TYPE-citation/edit/$PROJECT_ID/$CITATION_ID (edit existing citation)
   * /$STYLE/$TYPE-citation/new                           (edit new citation)
   */
  const projectID = getState().projectID || getUserBibInfo().projectId
  await dispatch(initProject(formPageParams.style, projectID))
  // If there is no citation id (new citation) this will return a promise as well
  await dispatch(getCitation(projectID, backendUser.citationId))
  dispatch(initFormForPage(formPageParams.type, formPageParams.style))
}

export const initLegacyFormJS = () => (dispatch, getState) => {
  const { citation } = getState()
  const detail = { ...citation, callback: () => {}, tab: window.gon.tabId }
  // legacy form js exposes a method that allows us to prepopulate
  // some of the form fields with data
  window.gon.fillForm({ detail })
}

export const formSubmit = () => (dispatch, getState) => {
  const { citation } = getState()
  const citationLink = citation.links ? citation.links.me : ''
  // legacy form js exposes a function on gon.submitForm.
  // we execute that function here, piping in data to it.
  window.gon.submitForm({
    detail: {
      callback: citationData => {
        if (citationLink !== '') {
          dispatch(updateCitation(citationLink, citationData))
        } else {
          dispatch(
            createCitation({
              citationData,
              manual: true,
            })
          )
        }
      },
    },
  })
}
