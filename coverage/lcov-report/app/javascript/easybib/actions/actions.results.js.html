const dataKeys = data => Object.keys(data).map(key => data[key])

/**
 * @param {string} url
 * @return {boolean}
 */
const isUrl = term => /^(http:\/\/|https:\/\/|www\.)/.test(term)

// https://github.com/easybib/sbm-issues/issues/1656
export const getExtraSearchParameters = type => {
  const types = {
    bible: 'bible',
    dissertation: 'dissertation',
    dissertationabstract: 'dissertationabstract',
    fedbill: 'fedbill',
    film: 'film',
    magazine: 'mags',
    map: 'map',
    musicalrecording: 'musicalrecording',
    newspaper: 'news',
    software: 'software',
    thesis: 'thesis',
  }
  return types[type] ? { srctype: types[type] } : {}
}

export const hasWorldcatLink = type =>
  [
    'bible',
    'book',
    'chapter',
    'conference',
    'dictionary',
    'dissertation',
    'dissertationabstract',
    'encyclopedia',
    'fedbill',
    'film',
    'map',
    'multivolume',
    'musicalrecording',
    'software',
    'thesis',
  ].includes(type)
 
export const addWorldcatLink = sources =>
  Object.values(sources).map(source => {
    const { oclc } = source.data.other
    const link = `${constants.WORLDCAT_URL}/${oclc}`
    return { link, ...source }
  })
 
/**
 * Used on the Search Results Page when someone selects a search result to cite.
 * @param {object} source     search result source.
 * @param {string} searchText search term text
 * @param {string} searchType type of thing searching for, eg. 'website', 'book'
 * @param {string} style      citation style, eg. 'mla8'
 *
 * Only book, journal and website citations display the Eval page.
 * Otherwise it goes to the form page.
 * @see https://github.com/easybib/sbm-issues/issues/2020
 */
export const useSearchResult = (source, searchText, searchType, style) => dispatch => {
  dispatch(storeSearchData({ source, searchText, searchType, style }))
  const page = ['book', 'journal', 'website'].includes(searchType) ? 'eval' : 'new'
  const url =
    page === 'new'
      ? `/${style}-format/${searchType}-citation/${page}`
      : `/${style}/${searchType}-citation/${page}`
  push(url)
}

/**
 * Search for sources.
 * Website search results need to use autocite to get the needed URL instead of bing search one.
 * @param {string} sourceText - searched term
 * @param {string} sourceType
 * @uses sourceApi.searchSource()
 */
export const searchSource = (sourceText, sourceType) => (
  dispatch,
  getState,
  { sourceApi }
) => {
  const alias = getExtraSearchParameters(sourceType)
  const uuid = dispatch(createLoading())
  const source = { text: sourceText, type: sourceType }
  dispatch(searchedSource(source))
  /**
   * For website and films: if searched term is an URL, use autocite
   */
  if (['film', 'website'].includes(sourceType) && isUrl(sourceText)) {
    return autociteUrl(sourceText).then(
      cslResponse => {
        sendMetrics(cslResponse, 'results', getState().user)
        const response = cslToEB(cslResponse)
        dispatch(loadSources(response))
        dispatch(loadingHide(uuid))
      },
      e => {
        // Do not allow citing image/media URL, redirect to manual website citation instead
        const autociteWrongType = 'This URL is a PDF, image or Office Document'
        if (e.response.data && e.response.data.includes(autociteWrongType)) {
          const style = getCitationStyle(getState())
          const url = `/${style}-format/${sourceType}-citation`
          return redirect(url)
        }
        dispatch(displayError(constants.CITATION_FROM_URL_ERROR, e))
        dispatch(loadSources([]))
        dispatch(loadingHide(uuid))
        return true
      }
    )
  }
  return sourceApi.searchSource(sourceText, sourceType, alias).then(
    resp => {
      const sources = hasWorldcatLink(sourceType)
        ? addWorldcatLink(resp.results)
        : resp.results
      dispatch(loadSources(sources))
      dispatch(loadingHide(uuid))
    },
    e => {
      dispatch(loadSources([]))
      dispatch(loadingHide(uuid))
      track(e)
    }
  )
}
 
/**
 * Journal sources have 3 search options: normal one, JSTOR and ProQuest.
 * @param {string} type
 */
export const searchJournal = type => (dispatch, getState) => {
  let url = ''
  const term = getState().search.text
  switch (type) {
    case 'jstor':
      url = `${constants.JSTOR_URL}&search=${term}`
      break
    case 'proquest':
      url = `${constants.PROQUEST_URL}&search=${term}`
      break
    default:
      dispatch(loadSources([]))
      return dispatch(searchSource(term, type))
  }
  dispatch(hideError())
  dispatch(loadSources([]))
  return request(url).then(
    resp => {
      dispatch(loadSources(resp.results))
    },
    e => {
      dispatch(loadSources([]))
      track(e)
    }
  )
}

export const initResultsPage = () => async (dispatch, getState) => {
  await dispatch(initApp('results'))
  const citationStyle = getCitationStyle(getState())
  const sourceType = getSourceType(getState())

  dispatch(searchSource(getParameterByName('q'), sourceType))
  // preload the form html, so it's in cache for when they land on the form page.
  requestForm(sourceType, citationStyle)
}
