/**
 * Used to be able to display a loading information while requests are pending.
 * Stored in state.loading.
 * @param {string} id - Unique id for that request
 * @return {string} - Unique id used to be able to hide the loading when that request resolves
 */
export const createLoading = (id?: string): EasybibThunkAction<string> => dispatch => {
  const uuid = id || v4()
  dispatch(loadingShow(uuid))
  return uuid
}

/**
 * Load all extra citation styles.
 * On Bibliography page, we only load "extra" styles, therefore MLA8 is missing.
 * @param {boolean} fixMissing - if true, add MLA8 in the beginning
 * @uses formatApi.getStyles()
 */
export const initializeDropdownStyles = (
  options = { fixMissing: false }
): EasybibThunkAction<void> => async (dispatch, getState, { formatApi }) => {
  const uuid = createLoading()
  const data = await formatApi.getStyles()
  const styles = data.results
  // There are 2 flavors of the styles dropdown.
  // One, is used on the search widget. The other is used on the bibliography page,
  // for the citations header. For the dropdown on bibliography page, the
  // dropdown serves a dual purpose of showing the user what their currently chosen
  // style is (but only for the popular styles). This logic accounts for that need
  // on bibliography
  if (options.fixMissing) {
    styles.unshift(
      { id: 'mla8', name: 'MLA8', value: 'mla8' },
      { id: 'mla7', name: 'MLA7', value: 'mla7' },
      { id: 'apa', name: 'APA', value: 'apa' },
      { id: 'chicago', name: 'Chicago', value: 'chicago' }
    )
  }
  dispatch(loadStyles(styles))
  dispatch(loadingHide(uuid))
}

export const getUserObject = async () => {
  let user = getUserBibInfo()
  if (isEmpty(user) || !user.exp || user.exp < getCurrentTime()) {
    user = await fetchUser()
  }
  return user
}

const fetchToken = (projectID: string): EasybibThunkAction<Promise<EBUser>> => async (
  dispatch,
  getState,
  { tokenStorage }
) => {
  const user = await getUserObject()
  tokenStorage.setToken(user.oauth.accessKey)
  // if we have a project id from the url, we want to update the session
  // and local storage projectIds to keep them in sync
  if (projectID && user.projectId !== projectID) {
    await requestWithCredentials('/api/session_project', 'POST', {
      projectId: projectID,
    })
    // update session projectId server side
    user.projectId = projectID
  }
  // update clientside local storage projectID
  user.projectID = projectID || user.projectID || user.projectId
  setEbUser(user)
  if (user.projectID) dispatch(saveProjectID(user.projectID))
  return user
}

export const verifyUrlStyle = style => (dispatch, getState) => {
  const isPro = userIsPro(getState().user.role)
  let verifiedStyle = style

  if (!isPro && !style.startsWith('mla')) {
    verifiedStyle = 'mla8'
    dispatch(showProPayWall())
  }
  return verifiedStyle
}
 
const hydrateCitationStyle = style => dispatch => {
  const verifiedStyle = dispatch(verifyUrlStyle(style || 'mla8'))
  dispatch(setCitationStyle(verifiedStyle))
}

const hydrateSourceType = (
  typeInUrl: string,
  typeFromRails: string,
  localBibInfo: any
): EasybibThunkAction<void> => dispatch => {
  const chosenSourceType = typeInUrl || typeFromRails || localBibInfo.type || 'website'
  setLocal('type', chosenSourceType)
  dispatch(setSourceType(chosenSourceType))
}
 
/**
 * Handle some conditions like errors, and redirects.
 * @param  {object} resp from getDataFromAPI
 */
export const handleApiDataErrors = (resp): EasybibThunkAction<void> => dispatch => {
  if (resp.apiError) {
    dispatch(displayError(constants.HOME_API_ERROR, resp))
    throw new Error(constants.HOME_API_ERROR)
  } else if (resp.redirect_to) {
    redirect(resp.redirect_to)
    return Promise.reject()
  }
  return Promise.resolve()
}
 
export const hydrateCell = cell => ({
  type: ActionTypes.CELL_SET,

})
 
export const hydrateFromServer = (): EasybibThunkAction<void> => async dispatch => {
  const page = getCurrentPage()
  dispatch(hydrateCitationStyle(urlParams(page).style))
  dispatch(hydrateSourceType(urlParams(page).type, window.gon.type, getLocalBibInfo()))
  dispatch(hydrateCell(window.gon.cell))
  dispatch(initializeDropdownStyles({ fixMissing: page === 'bibliography' }))
}
 
export const initUser = (page?: string): EasybibThunkAction<void> => async dispatch => {
  const { projectID } = urlParams(page)
  const resp = await dispatch(fetchToken(projectID))
  await dispatch(handleApiDataErrors(resp))
  dispatch(loginUser(resp.user, resp.loggedIn))
  dispatch(hydrateFromServer())
}

export const initApp = page => async dispatch => {
  await dispatch(initUser(page))
  return dispatch(setCitationCount())
}
 
/**
 * Request form HTML
 * Some of the forms have different names on the form api
 * @param {String} type - citation type
 * @param {String} style - citation style
 * @return {Promise}
 * @see https://github.com/easybib/sbm-issues/issues/1524#issuecomment-324594005
 */
export const requestForm = (type: string, style: string) => {
  const citationType = citationTypes.find(el => el.value === type)
  const requestUrl = `/api/form/${citationType.apiAlias}/${aliasToCode(style)}`
  return window.location.host.indexOf('playground.') === 0
    ? requestWithCredentials(requestUrl)
    : request(requestUrl)
}
