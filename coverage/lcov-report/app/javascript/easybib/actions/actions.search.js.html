/**
 * Actions involving saving search results for other pages.
 */
import ActionTypes from './actions.types'
 
import { redirect, redirectToForm } from '../actions/actions.redirects'
import { setSourceType } from '../actions/actions.shared'

export const checkSourceType = type => async (dispatch, getState) => {
  const selectedType = getState().types.find(stateType => stateType.value === type)
  if (selectedType.value === 'upload') {
    return redirect('/impexport/index/status')
  }
  if (selectedType.manual) {
    await dispatch(setSourceType(type))
    return dispatch(redirectToForm(''))
  }
  return dispatch(setSourceType(type))
}
 
/**
 * Used in EvalPage, FormPage to load a citation from a storage.
 * Uses sessionStorage underneath so it can share data between pages
 *   on the same window/tab.
 */
export const loadSearchData = () => ({
  type: ActionTypes.SEARCH_DATA_LOAD,
  searchData: JSON.parse(sessionStorage.getItem('ebSearchData') || '{}'),
})

/**
 * Used on Search Result Page to store a user selected search result they want to cite.
 */
export const storeSearchData = searchData => {
  sessionStorage.setItem('ebSearchData', JSON.stringify(searchData))
  return {
    type: ActionTypes.SEARCH_DATA_STORE,
    searchData,
  }
}
 
/**
 * Used in EvalPage, FormPage and BibliographyPage to know which citation is "active".
 * In the latter, this is the citation active when clicked on Inline-Citation, Footnote, Copy, etc
 * Stored in state.citation.
 * @param {object} citation
 */
export const loadCitation = citation => ({
  type: ActionTypes.CITATION_LOAD,
  citation,
})
